/** Default maximum request body size (bytes) the body parser accepts before a 413. */
export const DEFAULT_BODY_LIMIT = 1_048_576

/**
 * Default maximum DECOMPRESSED request body size (bytes) — the zip-bomb cap fed to
 * `node:zlib` as `maxOutputLength` when transparently decompressing a `Content-Encoding`
 * request body.
 *
 * @remarks
 * A small compressed payload (under {@link DEFAULT_BODY_LIMIT} on the wire) can inflate by
 * many orders of magnitude — a classic decompression bomb. Capping the codec's
 * `maxOutputLength` makes it ABORT (throw `RangeError` / `ERR_BUFFER_TOO_LARGE`) the moment
 * the output would exceed this, BEFORE the full bomb is materialized, so the inflation can
 * never exhaust memory; the body parser maps that abort to a `413`. Defaults to `16 MiB` — a
 * generous ceiling for a legitimate compressed JSON / form body, far below an OOM. A consumer
 * raises it via {@link import('./types.js').BodyParserOptions} `decompression` for a workload
 * that genuinely ships larger decompressed bodies.
 *
 * This constant governs ONLY the fallback cap on a bare `context.body()` decompression when NO
 * body-parser middleware is mounted. {@link import('./middlewares.js').createBodyParser} itself
 * defaults its `decompression` to its own `limit` (NOT this constant — "limit means limit"), so a
 * mounted parser's decompressed cap tracks its body `limit`, never this fallback.
 */
export const DEFAULT_DECOMPRESSED_LIMIT = 16_777_216

/**

 * part's REAL MIME type from (anti-spoofing) — keyed by MIME type, each value a list of candidate

 *
 * @remarks
 * Each signature is a sequence of bytes matched against the file's leading bytes; a `null` entry is a

/**
 * The default `Strict-Transport-Security` directive {@link import('./middlewares.js').createSecurity}
 * sets when HSTS is ENABLED — one year `max-age` with `includeSubDomains`.
 *
 * @remarks
 * HSTS is OFF by default (it is meaningful only over HTTPS and is browser-PINNED, so a wrong value is

 */
export const DEFAULT_HSTS = 'max-age=31536000; includeSubDomains'

/**
 * The default `Content-Security-Policy` directive {@link import('./middlewares.js').createSecurity}
 * sets — `default-src 'self'` PLUS four hardening directives:
 * `"default-src 'self'; base-uri 'self'; object-src 'none'; frame-ancestors 'self'; form-action 'self'"`.
 *
 * @remarks
 * `default-src 'self'` is the safe baseline (load resources only from the page's own origin); the
 * four added directives close injection / clickjacking gaps `default-src` alone does NOT cover:
 *
 * - **`base-uri 'self'`** — forbids an injected `<base href>` from re-pointing every relative URL on the
 *   page at an attacker origin (the dangling-markup / base-tag injection vector).
 * - **`object-src 'none'`** — blocks `<object>` / `<embed>` plugin content (a legacy XSS / data-exfil
 *   surface `default-src` does not reach in older parsers).
 * - **`frame-ancestors 'self'`** — the modern clickjacking defense (who may frame THIS page), which
 *   SUPERSEDES `X-Frame-Options` in CSP-aware browsers (the spine still also sets `X-Frame-Options`
 *   for older clients via {@link DEFAULT_FRAME_OPTIONS}).
 * - **`form-action 'self'`** — restricts where a `<form>` may POST, so an injected form cannot exfil
 *   credentials to an attacker endpoint.
 *
 * Most apps tighten / extend it for their CDN + inline-script needs via `createSecurity`'s `csp`
 * option — a `string` REPLACES this default WHOLESALE (it is not merged), `false` omits the header.
 */
export const DEFAULT_CSP =
	"default-src 'self'; base-uri 'self'; object-src 'none'; frame-ancestors 'self'; form-action 'self'"

/**

 *
 * @remarks

 * for an OAuth popup flow) or omits the header via `createSecurity`'s `coop` option (`false`).
 */
export const DEFAULT_COOP = 'same-origin'

/**
 * The default `Cross-Origin-Resource-Policy` value {@link import('./middlewares.js').createSecurity}
 * sets — `'same-origin'`.
 *
 * @remarks
 * CORP declares WHO may embed this response as a subresource: `same-origin` blocks ANOTHER origin from
 * `<img>` / `<script>` / `fetch`-loading it (a defense against cross-origin resource inclusion +
 * Spectre-style speculative reads of the response bytes). A consumer that serves PUBLIC cross-origin
 * assets (a CDN, a shared API) relaxes it to `'cross-origin'` (or `'same-site'`) — or omits the header
 * — via `createSecurity`'s `corp` option (`false`).
 *
 * NOTE: this is NOT `Cross-Origin-Embedder-Policy` (`require-corp`). COEP is OFF by default (it breaks
 * cross-origin subresources for most apps) and is exposed only as the opt-IN {@link
 * import('./types.js').SecurityOptions} `coep` flag — `createSecurity` never sets it unless asked.
 */
export const DEFAULT_CORP = 'same-origin'

/**

 * when the opt-IN `coep` option is enabled — `'require-corp'`.
 *
 * @remarks

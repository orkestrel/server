import type { MultipartReason } from './types.js'

// AGENTS ┬¦12: a handler signals a client-facing fault by throwing an `HTTPError`
// carrying the HTTP `status` to send; the error-boundary middleware turns it into a
// response of that status. Any OTHER throw is a programmer/runtime error ŌåÆ a 500
// (its message hidden unless `expose`). Mirrors the database layer's
// `DatabaseError` code/context shape, one level adapted: the machine-readable field
// here is the numeric `status` (plus an optional `context` bag).

/**
 * An error a handler (or middleware) throws to produce an HTTP response of a
 * specific status.
 *
 * @remarks
 * Carries the HTTP `status` to send and an optional `context` record (the offending
 * field / value). The error-boundary middleware ({@link createErrorBoundary})
 * catches it and writes a response of `status` with the error's `message` as the
 * body ŌĆö so an `HTTPError`'s message is ALWAYS client-facing (it is the handler's
 * deliberate signal), unlike a generic throw whose message is hidden. Subclass it
 * (as {@link ContentTooLargeError} does) for specific statuses, or throw it directly
 * (`throw new HTTPError(404, 'user not found')`). Narrow a caught value with
 * {@link isHTTPError}.
 *
 * @example
 * ```ts
 * import { HTTPError } from '@src/server'
 *
 * const handler = async (context) => {
 * 	const user = await find(context.params.id)
 * 	if (user === undefined) throw new HTTPError(404, 'user not found')
 * 	context.json(user)
 * }
 * ```
 */
export class HTTPError extends Error {
	readonly status: number
	readonly context?: Readonly<Record<string, unknown>>

	constructor(status: number, message: string, context?: Readonly<Record<string, unknown>>) {
		super(message)
		this.name = 'HTTPError'
		this.status = status
		this.context = context
	}
}

/**
 * The {@link HTTPError} thrown when a request body exceeds the body parser's size
 * limit ŌĆö a `413 Content Too Large`.
 *
 * @remarks
 * Thrown by {@link createBodyParser} (and the context's `body()` collection) when
 * the received bytes exceed the configured `limit`, carrying the `limit` in its
 * `context`. The error boundary renders it as a 413.
 */
export class ContentTooLargeError extends HTTPError {
	constructor(limit: number) {
		super(413, `request body exceeds the ${limit}-byte limit`, { limit })
		this.name = 'ContentTooLargeError'
	}
}

/**
 * The {@link HTTPError} {@link import('./middlewares.js').createMultipart} throws when a
 * `multipart/form-data` body breaches a limit, is malformed, or carries a rejected file ŌĆö after it
 * has CLEANED UP every already-staged temp file.
 *
 * @remarks
 * It subclasses {@link HTTPError} so the error boundary renders it by `status` like any other (a
 * `413` for a limit breach, `400` for a malformed body, `415` for a rejected type). The `reason`
 * discriminant ({@link MultipartReason}) names which fault it was, for a consumer that wants to
 * distinguish them (narrow with {@link isMultipartError}); the offending limit / type rides the
 * inherited `context`. Cleanup of the staged temp files is the THROWER's responsibility (done before
 * the throw), so catching this never leaks a temp file.
 */
export class MultipartError extends HTTPError {
	readonly reason: MultipartReason

	constructor(
		status: number,
		message: string,
		reason: MultipartReason,
		context?: Readonly<Record<string, unknown>>,
	) {
		super(status, message, context)
		this.name = 'MultipartError'
		this.reason = reason
	}
}

/**
 * Narrow an unknown caught value to an {@link HTTPError} (including its subclasses).
 *
 * @param value - The value to test (typically a `catch` binding)
 * @returns `true` when `value` is an {@link HTTPError}
 *
 * @example
 * ```ts
 * try {
 * 	await handle(context)
 * } catch (error) {
 * 	if (isHTTPError(error)) context.error(error.status, error.message)
 * }
 * ```
 */
export function isHTTPError(value: unknown): value is HTTPError {
	return value instanceof HTTPError
}

/**
 * Narrow an unknown caught value to a {@link MultipartError} ŌĆö a multipart-specific fault (a limit
 * breach, a malformed body, or a rejected file type).
 *
 * @param value - The value to test (typically a `catch` binding)
 * @returns `true` when `value` is a {@link MultipartError}
 */
export function isMultipartError(value: unknown): value is MultipartError {
	return value instanceof MultipartError
}

/**
 * Create an HTTP server — an observable lifecycle over `node:http` composing a route
 * manager, a Koa-onion middleware chain, and the core L1 `Abort` / `Timeout`.
 *
 * @remarks
 * Configure it with `use` (middleware, outer-first), `route` (one route or a batch),
 * and `group` (a prefixed registrar) before `start()`. `start()` listens on the
 * configured {@link ServerOptions.host} / {@link ServerOptions.port} — an omitted / `0`
 * port binds an EPHEMERAL OS-assigned port (the default) — and resolves the bound port;
 * `stop()` drains in-flight requests up to the `drain` deadline then closes; `destroy()`
 * tears it down. It guards every unhandled handler throw into a 500, so the process never
 * crashes on an escaping error — pair it with {@link
 * import('./middlewares.js').createErrorBoundary} (the outermost middleware) for structured
 * error responses.
 *
 * @param options - Optional `on` ({@link import('./types.js').ServerEventMap} initial
 *   listeners), `host` / `port` (the bind interface + port; omitted / `0` port ⇒ ephemeral
 *   — use {@link import('./helpers.js').discoverPort} to pre-pick a free one), and `drain`
 *   (graceful-stop deadline in ms); see {@link ServerOptions}
 * @returns A working {@link ServerInterface}
 *
 * @example
 * ```ts
 * import { createBodyParser, createCors, createErrorBoundary, createServer } from '@src/server'
 *
 * const server = createServer()
 * server.use(createErrorBoundary()) // outermost — catches downstream throws
 * server.use(createCors())
 * server.use(createBodyParser())
 * server.route({
 * 	method: 'POST',
 * 	path: '/echo',
 * 	handler: async (context) => context.json(await context.body()),
 * })
 *
 * const port = await server.start() // ephemeral port
 * // ... fetch(`http://127.0.0.1:${port}/echo`, { method: 'POST', body: '{"hi":1}' })
 * await server.stop() // graceful drain, then close
 * ```
 */
export function createServer(options?: ServerOptions): ServerInterface {
	return new Server(options)
}

/**
 * Create an in-memory {@link SessionStoreInterface} — the DEFAULT session store {@link
 * import('./middlewares.js').createSession} builds when no `store` is supplied, and a
 * ready-made store to SHARE across several session middlewares.
 *
 * @remarks
 * A thin wrapper over {@link import('./stores/MemorySessionStore.js').MemorySessionStore}: a
 * process-local `Map` of sessions with lazy idle-TTL eviction (the lifted closure store the
 * session middleware used to own inline). The store OWNS the idle-TTL policy — `get` applies
 * the expiry sweep + the touch-on-access — so the session value stays pure identity. The
 * session analogue of {@link createMemoryQueueStore}, but LEANER (only `get` / `set` /
 * `delete`). For a DURABLE store (sessions surviving a restart), implement {@link
 * SessionStoreInterface} over a database / cache and pass it as `createSession`'s `store`.
 *
 * @typeParam S - The concrete session type the store holds (defaults to {@link Session})
 * @param options - Optional `ttl` (the session idle lifetime in ms); see {@link
 *   MemorySessionStoreOptions}
 * @returns A memory-backed {@link SessionStoreInterface}, typed by `S`
 *
 * @example
 * ```ts
 * import { createMemorySessionStore, createSession } from '@src/server'
 *
 * const store = createMemorySessionStore({ ttl: 60_000 })
 * server.use(createSession({ store })) // an explicit store (shareable across middlewares)
 * ```
 */
export function createMemorySessionStore<S extends SessionInterface = Session>(
	options?: MemorySessionStoreOptions,
): SessionStoreInterface<S> {
	return new MemorySessionStore<S>(options)
}

/**
 * Create a {@link DatabaseSessionStore} over any {@link DriverInterface} — the durable,
 * driver-pluggable backing for the session seam, the opt-in twin of {@link
 * createMemorySessionStore} that REPLICATES its exact idle-TTL + touch-on-access contract.
 *
 * @remarks
 * Builds a one-table database (`sessions`, keyed by `id`) over the supplied driver, with the
 * `{ id; session; lastSeen }` ({@link SessionRow}) column map — `session` is ONE OPAQUE JSON COLUMN
 * (a `rawShape`, exactly as {@link createDatabaseQueueStore} stores its `input`) and `lastSeen` a
 * `number` column holding the store-owned idle-TTL cursor. The opaque column keeps the row type
 * FLAT for the OPEN generic `S` (a typed-`S` column would force a TS2589-tripping inference), so the
 * session is read back as the broad `unknown` and narrowed to a session on `get` by the floor guard
 * {@link import('./helpers.js').isSession} (the AGENTS §14 boundary narrow — never a cast). The
 * `driver` DEFAULTS to {@link createMemoryDriver}, so the store ALSO works in memory out of the box;
 * pass a server `createSQLiteDriver` (or a browser IndexedDB driver) for a persistent one — the
 * durability is the driver's job, the store engine is shared, and it swaps in behind {@link
 * SessionStoreInterface} WITHOUT touching {@link import('./middlewares.js').createSession} (the store
 * OWNS the idle-TTL, so the `ttl` reaches it here, exactly as the memory twin's does). A session
 * persisted to a real driver must be JSON-serializable (the same caveat the queue store's `input`
 * carries).
 *
 * @typeParam S - The concrete session type the store holds (defaults to {@link Session})
 * @param driver - The storage backend the sessions persist to (defaults to {@link createMemoryDriver})
 * @param ttl - The session IDLE lifetime in ms (the store evicts an idle session on access); omit
 *   for sessions that live until an explicit `delete`
 * @param absoluteTtl - The session ABSOLUTE lifetime in ms from mint (the store evicts a session this
 *   old even if continuously active, via the `createdAt` column); omit for idle-only bounding
 * @returns A {@link SessionStoreInterface} over the driver, typed by `S`
 *
 * @example
 * ```ts
 * import { createMemoryDriver } from '@src/core'
 * import { createDatabaseSessionStore, createSession } from '@src/server'
 *
 * const store = createDatabaseSessionStore(createMemoryDriver(), 60_000) // a durable driver swaps in here
 * server.use(createSession({ store })) // sessions now persist through the store
 * ```
 */
export function createDatabaseSessionStore<S extends SessionInterface = Session>(
	driver?: DriverInterface,
	ttl?: number,
	absoluteTtl?: number,
): SessionStoreInterface<S>
// The public overload types the store by `S`. The implementation runs on the BROAD floor
// (`SessionInterface`), so the `sessions` table is built and read ONCE — not re-instantiated per
// concrete `S` — over the OPAQUE `session` column (`rawShape`, infers `unknown`), which keeps the
// row type flat and sidesteps the TS2589 a typed-`S` column would spring. The floor guard
// `isSession` narrows the column back to `SessionInterface` on read (the §14 boundary narrow); no
// cast — the narrowed value satisfies the broad store's `S = SessionInterface`, and every call site
// still sees the precise `SessionStoreInterface<S>` from the overload.
export function createDatabaseSessionStore(
	driver: DriverInterface = createMemoryDriver(),
	ttl?: number,
	absoluteTtl?: number,
): SessionStoreInterface<SessionInterface> {
	const columns = {
		id: stringShape(),
		session: rawShape({}),
		lastSeen: integerShape({ min: 0 }),
		createdAt: integerShape({ min: 0 }),
	}
	const database = createDatabase({ driver, tables: { sessions: columns } })
	const table: TableInterface<SessionRow> = database.table('sessions')
	return new DatabaseSessionStore<SessionInterface>(table, isSession, ttl, absoluteTtl)
}

/**
 * Create a {@link NegotiatorInterface} — the reusable content-negotiation machine over the weighted
 * `Accept` / `Accept-Encoding` / `Accept-Language` family.
 *
 * @remarks
 * It is shared, cross-middleware machinery (a {@link Negotiator} entity), NOT a middleware (AGENTS
 * §5) — a route or middleware composes it. `negotiate` picks the client's preferred media type
 * (exact / subtype-wildcard / `*​/*`); `encoding` / `language` are the sibling axes; `format`
 * dispatches a {@link import('./types.js').FormatHandlerMap} to the preferred representation or
 * answers `406`. It reuses the SAME q-value parser ({@link import('./helpers.js').parseAcceptHeader})
 * {@link import('./middlewares.js').createCompression} uses — one parser, never two.
 *
 * @returns A working {@link NegotiatorInterface}
 *
 * @example
 * ```ts
 * import { createNegotiator, createServer } from '@src/server'
 *
 * const negotiator = createNegotiator()
 * const server = createServer()
 * server.route({
 * 	method: 'GET',
 * 	path: '/data',
 * 	handler: (context) =>
 * 		negotiator.format(context, {
 * 			'application/json': (c) => c.json({ ok: true }),
 * 			'text/plain': (c) => c.text('ok'),
 * 		}), // 406 if the client accepts neither
 * })
 * ```
 */
export function createNegotiator(): NegotiatorInterface {
	return new Negotiator()
}

/**
 * Create one staged-upload {@link UploadedFileInterface} — a FROZEN, immutable data record (the
 * sibling of {@link import('@src/core').createFile}), built by {@link
 * import('./middlewares.js').createMultipart} for each file part the streaming multipart parser
 * staged to a temp file.
 *
 * @remarks
 * The record is `Object.freeze`d, so it is immutable — its operations are FREE FUNCTIONS ({@link
 * import('./helpers.js').streamUploadedFile} / {@link import('./helpers.js').readUploadedFile} /
 * {@link import('./helpers.js').moveUploadedFile}), never methods on the value; `move` does not mutate
 * but RETURNS a NEW frozen record with `path` repointed and `status: 'moved'`. The `status` defaults
 * to `'staged'` (a freshly staged file), so the parser need not spell it out.
 *
 * @param input - The {@link UploadedFileInput} (`field` / `name` / `size` / `mime` / `validated` /
 *   `path`, plus an optional `status` defaulting to `'staged'`)
 * @returns A frozen {@link UploadedFileInterface}
 *
 * @example
 * ```ts
 * import { createUploadedFile, moveUploadedFile } from '@src/server'
 *
 * const file = createUploadedFile({
 * 	field: 'avatar', name: 'cat.png', size: 1024, mime: 'image/png', validated: true, path: '/tmp/xyz',
 * })
 * file.status // 'staged'
 * const moved = await moveUploadedFile(file, '/data/avatar') // a NEW record; moved.status === 'moved'
 * ```
 */
export function createUploadedFile(input: UploadedFileInput): UploadedFileInterface {
	return Object.freeze({
		field: input.field,
		name: input.name,
		size: input.size,
		mime: input.mime,
		validated: input.validated,
		status: input.status ?? 'staged',
		path: input.path,
	})
}

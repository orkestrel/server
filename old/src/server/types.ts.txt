import type { Session } from './Session.js'
import type { IncomingMessage, ServerResponse } from 'node:http'
import type { Duplex } from 'node:stream'
import type { EmitterErrorHandler, EmitterHooks, EmitterInterface } from '@src/core'

// The HTTP spine's public surface — the source of truth (AGENTS §2). A lean,
// production-shaped server over `node:http`: a `Server` lifecycle, a `RouteManager`
// (params / query / auto HEAD / OPTIONS / 404 / 405 / groups), a Koa-onion
// `MiddlewareManager`, and the single unified `RouteHandlerContext` every handler +
// middleware receives. It composes the core L1 primitives (`Abort` / `Emitter` /
// `Timeout`) and adds no new concurrency primitive. The ONE streaming surface is the
// GENERIC Server-Sent-Events seam (`openSSEStream` / `bindStreamingAbort` over the
// raw `ServerResponse` — agent-agnostic, so an MCP HTTP transport reuses it). The
// other generic seam is the `upgrade` hook: a raw `node:http` upgrade is fanned out to
// registered {@link UpgradeHandler}s (the first claimer owns the socket, an unclaimed
// one is destroyed) — the spine stays WebSocket-agnostic, so a WebSocket transport
// attaches WITHOUT the spine knowing the RFC 6455 protocol. NO auth baked in. The
// `node:http` `IncomingMessage` / `ServerResponse` (and the upgrade `Duplex` socket)
// are `import type` only and narrowed via guards at the boundary, never asserted
// (AGENTS §14).

/** An HTTP request method this server routes. */
export type HTTPMethod = 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE' | 'HEAD' | 'OPTIONS'

/**
 * The single unified surface every route handler and middleware receives — the
 * ambient state of one in-flight request/response exchange.
 *
 * @remarks
 * - `request` / `response` — the raw `node:http` message + response, for the rare
 *   case a handler needs them directly (status codes, streaming a file). Prefer the
 *   response helpers (`json` / `text` / `empty` / `error`) for the common cases.
 * - `method` — the request method narrowed to an {@link HTTPMethod} (an unknown
 *   verb is treated as `GET` at the boundary, never asserted).
 * - `url` — the parsed request {@link URL} (origin reconstructed from the `Host`
 *   header), so `url.pathname` / `url.searchParams` are ready without re-parsing.
 * - `params` — the path parameters the matched route extracted (`/users/:id` →
 *   `{ id }`), URL-decoded; an empty record for a parameterless route.
 * - `query` — the URL query string parsed once (`url.searchParams`).
 * - `body()` — LAZILY collect + JSON-parse the request body, cached so repeated
 *   calls (a body-parser middleware then the handler) read it ONCE. Resolves the
 *   parsed value (`undefined` for an empty body); a non-JSON body resolves the raw
 *   decoded text. The {@link Middleware} from `createBodyParser` enforces the size
 *   limit by driving this same single collection path.
 * - `state` — a mutable per-request bag for threading application dependencies (an
 *   agent manager, a database handle) from a middleware down to the handler.
 * - `json` / `text` / `empty` / `error` — response helpers that write the status +
 *   headers + body in one call (wrapping the pure `helpers.ts` senders); each is a
 *   no-op once the response has been sent, so a handler can't double-write.
 */
export interface RouteHandlerContextInterface {
	readonly request: IncomingMessage
	readonly response: ServerResponse
	readonly method: HTTPMethod
	readonly url: URL
	readonly params: Readonly<Record<string, string>>
	readonly query: URLSearchParams
	readonly state: Map<string, unknown>
	body(): Promise<unknown>
	json(value: unknown, status?: number): void
	text(value: string, status?: number): void
	empty(status?: number): void
	error(status: number, message?: string): void
}

/**
 * A route handler — runs the request's work against its {@link
 * RouteHandlerContextInterface}, writing the response through the context's helpers
 * (or the raw `response`). May be async; a throw is caught by the error-boundary
 * middleware (or the server's last-resort 500 guard).
 */
export type RouteHandler = (context: RouteHandlerContextInterface) => Promise<void> | void

/**
 * A middleware in the Koa onion — runs around the rest of the chain.
 *
 * @remarks
 * Receives the shared {@link RouteHandlerContextInterface} and a `next` thunk; call
 * `await next()` to invoke the downstream middleware (ending in the matched route's
 * handler), running code before and/or after it. Calling `next()` more than once
 * rejects (the double-`next` guard). Omitting `next()` short-circuits the chain.
 */
export type Middleware = (
	context: RouteHandlerContextInterface,
	next: () => Promise<void>,
) => Promise<void> | void

/**
 * A handler for an HTTP protocol-upgrade — the seam a transport that owns the raw
 * connection (a WebSocket transport) registers via {@link ServerInterface.upgrade}.
 *
 * @remarks
 * Invoked with the raw `node:http` upgrade triple — the `request` carrying the upgrade
 * headers (`Upgrade` / `Sec-WebSocket-Key` / …), the upgraded `node:stream` Duplex
 * `socket` (the raw TCP stream, ready for the handler to write the `101` handshake and
 * speak its own wire protocol), and `head` (any bytes already buffered after the
 * upgrade request, to replay through the handler's parser). RETURN `true` to CLAIM the
 * socket — the handler now OWNS it (no later handler runs, and the spine will not
 * destroy it); return `false`/falsy to DECLINE, so the next registered handler gets a
 * chance. A handler that THROWS is caught and treated exactly as a DECLINE — the throw is
 * surfaced on the {@link ServerEventMap}'s `error` event (the same channel an escaping
 * route-handler throw uses), and the fan-out continues to the next handler — so a buggy
 * handler can never crash the process or leak the socket (symmetric with the request
 * path). If NO handler claims it, the spine destroys the socket (an unhandled upgrade
 * must not leak a dangling connection). The spine stays protocol-agnostic — it fans the
 * raw upgrade out and never speaks WebSocket itself.
 */
export type UpgradeHandler = (request: IncomingMessage, socket: Duplex, head: Buffer) => boolean

/**
 * One Server-Sent Event to SERIALIZE to the wire — the inverse of the core
 * {@link import('@src/core').SSEEvent} a {@link import('@src/core').SSEParserInterface}
 * decodes. Field names are kept identical to that event so the encode ↔ decode
 * round-trip is obvious.
 *
 * @remarks
 * - `data` — the event payload (required). Serialized as one or more `data:` lines:
 *   the value is SPLIT on `\n` into a `data:` line PER segment, so it round-trips
 *   through the parser's multi-`data` concat (`'a\nb'` → `data: a` + `data: b` →
 *   `'a\nb'`). An empty string still emits a single empty `data:` line.
 * - `event` — the optional event TYPE (the parser's `event` field); emitted as an
 *   `event:` line. Omitted ⇒ the consumer's default (`message`).
 * - `id` — the optional last-event-id; emitted as an `id:` line for the consumer to
 *   track as reconnection state.
 * - `retry` — the optional reconnection time in milliseconds; emitted as a `retry:`
 *   line (an integer).
 */
export interface SSEMessage {
	readonly data: string
	readonly event?: string
	readonly id?: string
	readonly retry?: number
}

/**
 * Options for {@link openSSEStream}.
 *
 * @remarks
 * - `status` — the HTTP status the SSE response head is written with; defaults to
 *   `200`. (A streaming response is almost always a `200` — this is the rare escape
 *   hatch.)
 * - `headers` — extra response headers merged with the SSE headers the seam always
 *   sets (`Content-Type: text/event-stream`, `Cache-Control: no-cache`, `Connection:
 *   keep-alive`, and the proxy-buffering opt-out) — e.g. a CORS header. A key the seam
 *   owns is not overridden.
 */
export interface SSEStreamOptions {
	readonly status?: number
	readonly headers?: Readonly<Record<string, string>>
}

/**
 * A handle to write Server-Sent Events to an open HTTP response — the GENERIC
 * streaming surface {@link openSSEStream} returns over a {@link
 * RouteHandlerContextInterface}'s raw `response`.
 *
 * @remarks
 * Agent-agnostic by design: it speaks only the SSE wire vocabulary
 * ({@link SSEMessage}), so any streaming consumer (a live agent turn now, an MCP HTTP
 * transport later) maps its own events onto `write`. Each `write` serializes the
 * message to the wire (the `\n`→multi-`data:` split so it round-trips through the core
 * {@link import('@src/core').SSEParserInterface}) and flushes it; `comment` writes a
 * `: text` keep-alive line (which the parser ignores — no spurious event); `end` ends
 * the response. Every method is a SAFE NO-OP once `closed` is `true` (the response
 * finished or the client disconnected), so a late `write` never throws. Pair it with
 * {@link bindStreamingAbort} to map a client disconnect to a cancel.
 */
export interface SSEStreamInterface {
	/** Whether the underlying response is done (ended, or the connection closed). */
	readonly closed: boolean
	/**
	 * Serialize + flush one {@link SSEMessage} to the wire (a no-op once `closed`).
	 *
	 * @param message - The event to send (its `data` split on `\n` into `data:` lines)
	 */
	write(message: SSEMessage): void
	/**
	 * Write a `: text` SSE comment line — a keep-alive the parser ignores (no event).
	 *
	 * @param text - The comment text (sent after the `: ` prefix)
	 */
	comment(text: string): void
	/** End the response, completing the stream (a no-op once already `closed`). */
	end(): void
}

/**
 * The data needed to register one route.
 *
 * @remarks
 * - `method` — the {@link HTTPMethod} this route answers (a `GET` route also
 *   answers `HEAD` automatically).
 * - `path` — the path pattern, with `:name` segments captured as params
 *   (`/users/:id`); matched against the request pathname in full.
 * - `name` — an optional label for the route (tracing / reverse lookup).
 * - `handler` — the {@link RouteHandler} run on a match.
 */
export interface RouteInput {
	readonly method: HTTPMethod
	readonly path: string
	readonly name?: string
	readonly handler: RouteHandler
}

/** A registered route — a compiled, immutable {@link RouteInput}. */
export interface RouteInterface {
	readonly method: HTTPMethod
	readonly path: string
	readonly name: string | undefined
	readonly handler: RouteHandler
	/**
	 * Match a request pathname against this route's pattern.
	 *
	 * @param pathname - The request pathname to test
	 * @returns The extracted (URL-decoded) params on a match, else `undefined`
	 */
	match(pathname: string): Readonly<Record<string, string>> | undefined
}

/**
 * The outcome of resolving a request method + pathname against the registered
 * routes — a hit carries the route + its params; a miss distinguishes 405 (the
 * path matched some route, but not for this method — `allow` lists the methods that
 * do) from 404 (no route matched the path at all).
 */
export type RouteMatch =
	| {
			readonly status: 'matched'
			readonly route: RouteInterface
			readonly params: Readonly<Record<string, string>>
	  }
	| { readonly status: 'unmethoded'; readonly allow: readonly HTTPMethod[] }
	| { readonly status: 'unmatched' }

/**
 * A scoped registrar that prepends a path prefix — returned by {@link
 * RouteManagerInterface.group} so a set of routes shares a common base path
 * without repeating it. Its `add` (and nested `group`) forward to the parent with
 * the prefix applied.
 */
export interface RouteGroupInterface {
	readonly prefix: string
	add(input: RouteInput): void
	add(inputs: readonly RouteInput[]): void
	group(prefix: string): RouteGroupInterface
}

/**
 * The route registry — registers routes and resolves a request method + pathname to
 * the matched route (or a 404 / 405 signal), with automatic HEAD-from-GET and the
 * OPTIONS `Allow` set.
 */
export interface RouteManagerInterface {
	readonly count: number
	add(input: RouteInput): void
	add(inputs: readonly RouteInput[]): void
	/**
	 * Resolve a request to a route + params, or the miss kind (405 vs 404).
	 *
	 * @remarks
	 * A `HEAD` request matches a `GET` route (the handler runs; the spine discards
	 * the body). The `unmethoded` (405) result's `allow` is the set of methods the
	 * pathname does answer — for the `Allow` header.
	 *
	 * @param method - The request {@link HTTPMethod}
	 * @param pathname - The request pathname
	 * @returns The {@link RouteMatch}
	 */
	match(method: HTTPMethod, pathname: string): RouteMatch
	/** The set of methods registered for a pathname — for an OPTIONS `Allow`. */
	allow(pathname: string): readonly HTTPMethod[]
	routes(): readonly RouteInterface[]
	/**
	 * Open a scoped registrar that prepends `prefix` to every route added through
	 * it (and through nested groups).
	 *
	 * @param prefix - The path prefix to prepend (e.g. `/api`)
	 * @returns A {@link RouteGroupInterface} over this manager
	 */
	group(prefix: string): RouteGroupInterface
	clear(): void
}

/**
 * The middleware chain — registers middleware in order and composes them into one
 * onion dispatcher ending in a final handler.
 */
export interface MiddlewareManagerInterface {
	readonly count: number
	use(middleware: Middleware): void
	/**
	 * Compose the registered middleware into a single runner that threads `context`
	 * through the onion and ends in `handler`.
	 *
	 * @remarks
	 * The returned runner invokes `middleware[0](context, () => middleware[1](…))`
	 * down to `handler`. A middleware that calls its `next()` more than once causes
	 * the runner to reject (the double-`next` guard), so the chain runs each link at
	 * most once.
	 *
	 * @param handler - The terminal handler the onion wraps (the route dispatch)
	 * @returns A runner that executes the chain for one context
	 */
	compose(handler: RouteHandler): (context: RouteHandlerContextInterface) => Promise<void>
	clear(): void
}

/** The lifecycle state of a {@link ServerInterface} (AGENTS §10 vocabulary). */
export type ServerStatus = 'idle' | 'starting' | 'listening' | 'stopping' | 'stopped'

/**
 * The {@link ServerInterface}'s observable events (AGENTS §13).
 *
 * @remarks
 * Each event is a single noun/verb; tuple elements are labeled for IDE hover.
 * `upgrade` carries the upgrade request plus whether a registered {@link
 * UpgradeHandler} claimed the socket (`handled`) — fired AFTER the fan-out resolves,
 * for observability. `error` is a DOMAIN event (a server-level fault). Listener isolation
 * is the emitter's (AGENTS §13): a listener throw is routed to the emitter's `error` handler
 * (the `error` option), never onto this map, so a throwing observer never escapes into the
 * request/lifecycle engine.
 */
export type ServerEventMap = {
	start: [port: number]
	stop: []
	request: [method: HTTPMethod, url: string]
	upgrade: [request: IncomingMessage, handled: boolean]
	error: [error: unknown]
	drain: [pending: number]
}

/**
 * Options for `createServer`.
 *
 * @remarks
 * - `on` — the reserved {@link EmitterHooks} for the {@link ServerEventMap} (AGENTS
 *   §8), wiring initial lifecycle listeners at construction.
 * - `host` — the network interface `start()` binds to (the `node:http`
 *   `server.listen` host). Omitted ⇒ node's default (all available interfaces — an
 *   unspecified `0.0.0.0` / `::`); set it to `'127.0.0.1'` to bind loopback only.
 * - `port` — the TCP port `start()` binds to. Omitted (or `0`) ⇒ an EPHEMERAL,
 *   OS-assigned free port (the default — every shipped server stands on one); set a
 *   fixed port to bind it. `start()` still resolves + returns the ACTUALLY bound port
 *   (an ephemeral `0` resolves to the OS-assigned one). A port already in use rejects
 *   `start()` with the `node:http` `EADDRINUSE` — use {@link discoverPort} to pick a
 *   guaranteed-free port first.
 * - `drain` — the graceful-stop deadline in milliseconds: on `stop()` the server
 *   stops accepting new requests and hands in-flight requests this long to finish
 *   before forcing the socket closed. Defaults to a built-in grace.
 */
export interface ServerOptions {
	readonly on?: EmitterHooks<ServerEventMap>
	/** The emitter's listener-error handler (AGENTS §13) — a listener throw routes here, not to a domain event. */
	readonly error?: EmitterErrorHandler
	readonly host?: string
	readonly port?: number
	readonly drain?: number
}

/**
 * The HTTP server facade — an observable lifecycle over `node:http` composing a
 * route manager, a middleware chain, the L1 `Abort` (stop signal) and `Timeout`
 * (drain grace).
 *
 * @remarks
 * `use` / `upgrade` / `route` / `group` configure the server before `start`; `start`
 * listens on the configured {@link ServerOptions.host} / {@link ServerOptions.port}
 * (an omitted / `0` port → an EPHEMERAL port, the default, resolved from the bound
 * address) and resolves the bound port; `stop` drains in-flight requests up to the
 * `drain` deadline then closes; `destroy` tears everything down. Per request the spine builds a {@link
 * RouteHandlerContextInterface}, runs the composed middleware onion (ending in the
 * matched route's handler or a 404 / 405), and guards every unhandled throw so a
 * handler error becomes a 500 rather than crashing the process.
 */
export interface ServerInterface {
	readonly id: string
	readonly status: ServerStatus
	readonly emitter: EmitterInterface<ServerEventMap>
	readonly port: number | undefined
	use(middleware: Middleware): void
	upgrade(handler: UpgradeHandler): void
	route(input: RouteInput): void
	route(inputs: readonly RouteInput[]): void
	group(prefix: string): RouteGroupInterface
	start(): Promise<number>
	stop(): Promise<void>
	destroy(): Promise<void>
}

/**
 * Options for `createBodyParser` — the JSON body-parsing middleware.
 *
 * @remarks
 * - `limit` — the maximum request body size in bytes; a larger body throws a
 *   {@link ContentTooLargeError} (413). Defaults to a built-in cap.
 * - `decompression` — the maximum DECOMPRESSED body size in bytes (the zip-bomb cap)
 *   for a `Content-Encoding: gzip` / `deflate` / `br` request body. A highly-
 *   compressible payload that is small ON THE WIRE (under `limit`) can inflate
 *   enormously, so `node:zlib` is given this as its `maxOutputLength` — it ABORTS the
 *   inflation the instant the output would exceed the cap (it never allocates the full
 *   bomb), and the body parser maps that to a `413`. Defaults to ALIGN WITH `limit` (so
 *   a tight `limit` bounds the decompressed body too — "limit means limit"); set it
 *   independently — and larger — to allow a decompressed body that legitimately inflates
 *   above the compressed-byte `limit`. A NON-POSITIVE value (`0`) means UNCAPPED
 *   decompressed output (no `maxOutputLength` is fed to `node:zlib`) — use ONLY when the
 *   compressed input is already bounded by `limit`, since an uncapped inflation forfeits the
 *   zip-bomb guard.
 */
export interface BodyParserOptions {
	readonly limit?: number
	readonly decompression?: number
}

/**
 * Options for `createErrorBoundary` — the outermost error-catching middleware.
 *
 * @remarks
 * - `expose` — when `true`, a non-{@link HTTPError} throw's message is sent in the
 *   500 response body; otherwise the body is a generic message and the real error
 *   stays server-side. Defaults to `false`. (An {@link HTTPError}'s own message is
 *   always sent — it is the handler's intentional client-facing error.)
 * - `report` — a fire-and-forget sink the caught error is handed to (logging /
 *   metrics); its own throw is swallowed so reporting can't crash the response.
 */
export interface ErrorBoundaryOptions {
	readonly expose?: boolean
	readonly report?: (error: unknown) => void
}

/**
 * Options for `createCors` — the cross-origin middleware.
 *
 * @remarks
 * - `origin` — the allowed origin(s): a single origin string, a list (echoed back
 *   when the request `Origin` is in it), or `'*'`. Defaults to `'*'`.
 * - `methods` — the methods advertised in a preflight `Access-Control-Allow-Methods`.
 *   Defaults to the common set.
 * - `headers` — the request headers advertised in a preflight
 *   `Access-Control-Allow-Headers`. Defaults to `Content-Type` + `Authorization`.
 */
export interface CorsOptions {
	readonly origin?: string | readonly string[]
	readonly methods?: readonly HTTPMethod[]
	readonly headers?: readonly string[]
}

/**
 * A secret (or rotation list) for signing + verifying the spine's stateless tokens.
 *
 * @remarks
 * A single `string` is the lone secret. A `readonly string[]` is a rotation list in
 * `[current, ...older]` order: `signToken` always signs with the FIRST (the current
 * secret), while `verifyToken` (and `createTokenGuard`) accept a token signed by ANY
 * secret in the list — so a key can be rotated by prepending the new one and keeping
 * the old until every outstanding token has expired, with zero downtime.
 */
export type TokenSecret = string | readonly string[]

/**
 * Options for `signToken` — how a stateless, HMAC-signed token is minted.
 *
 * @remarks
 * - `secret` — the {@link TokenSecret} to sign with; `signToken` always uses the
 *   FIRST secret (a single string, or the current head of a rotation list). An empty
 *   rotation list is a misconfiguration and throws (sign is the one non-total path —
 *   verify stays total).
 * - `ttl` — an optional lifetime in milliseconds. When set, `signToken` binds an
 *   expiry timestamp (`Date.now() + ttl`) INTO the signed payload, so the expiry is
 *   HMAC-covered and cannot be tampered with; `verifyToken` then rejects the token
 *   once that instant has passed. Omitted ⇒ the token NEVER expires — and because the
 *   scheme is stateless there is no per-token revocation, so the ONLY way to invalidate
 *   an issued token is to rotate the secret, which invalidates EVERY outstanding token
 *   at once. Always set a `ttl` for any bearer that travels to a browser, so a leaked
 *   token self-expires.
 */
export interface TokenOptions {
	readonly secret: TokenSecret
	readonly ttl?: number
}

/**
 * Options for `createTokenGuard` — the stateless signed-token authentication
 * middleware.
 *
 * @remarks
 * - `secret` — the {@link TokenSecret} every incoming token is verified against. A
 *   rotation list accepts a token signed by any of its secrets (see {@link
 *   TokenSecret}).
 * - `header` — the request header carrying the token; defaults to {@link
 *   DEFAULT_TOKEN_HEADER} (`'authorization'`). Compared lower-cased (node lower-cases
 *   incoming header names).
 * - `scheme` — the auth-scheme prefix stripped before verifying; defaults to {@link
 *   DEFAULT_TOKEN_SCHEME} (`'Bearer'`), matched case-insensitively (`Bearer <token>`).
 *   Set it to `''` to treat the whole header value as the raw token (no scheme).
 */
export interface TokenGuardOptions {
	readonly secret: TokenSecret
	readonly header?: string
	readonly scheme?: string
}

/**
 * Options for `createRateLimiter` — the per-client fixed-window rate-limit middleware.
 *
 * @remarks
 * - `max` — the maximum number of requests one client key may make within a window.
 *   The window allows exactly `max` (the over-limit request is the `max + 1`th, which
 *   is rejected with a `429`).
 * - `window` — the window length in milliseconds. Each key's window is a fixed span
 *   that rolls LAZILY: the first request after a window has elapsed opens a fresh one
 *   (there is no background timer — a middleware factory has no teardown hook).
 * - `capacity` — the maximum number of distinct client keys retained at once, bounding
 *   memory. Reaching it before inserting a brand-new key evicts the oldest-inserted
 *   one. Defaults to {@link DEFAULT_RATE_LIMIT_CAPACITY}.
 * - `key` — the client-key extractor: maps a request's {@link
 *   RouteHandlerContextInterface} to the string the limit is counted against. Defaults
 *   to the verified token (when an upstream {@link Middleware} from `createTokenGuard`
 *   stashed one under {@link TOKEN_STATE}) else the raw socket peer IP — NOT a
 *   forwardable header (`X-Forwarded-For` is spoofable; a deployment behind a trusted
 *   proxy supplies its own `key`).
 * - `message` — the body message sent in the `429` response. Defaults to {@link
 *   DEFAULT_RATE_LIMIT_MESSAGE}.
 * - `clock` — the `() => number` epoch-ms clock the middleware reads for ALL window math
 *   (each window's reset instant, the lazy roll check, the `Retry-After` seconds);
 *   defaults to `Date.now`. The deterministic clock seam ({@link SessionOptions}'
 *   sibling): a test injects a manual clock and drives a window roll by advancing it
 *   explicitly — never racing a real window against wall-clock (AGENTS §16). Production
 *   never sets it.
 */
export interface RateLimiterOptions {
	readonly max: number
	readonly window: number
	readonly capacity?: number
	readonly key?: (context: RouteHandlerContextInterface) => string
	readonly message?: string
	readonly clock?: () => number
}

/**
 * The minimal shape of one server session — the base every {@link createSession}-managed
 * session satisfies.
 *
 * @remarks
 * Deliberately tiny: a session is identified by its `id` and nothing more. The session
 * store ({@link SessionStoreInterface}) tracks each session's `lastSeen` instant ITSELF
 * for the lazy idle-TTL sweep, so the session value carries no timekeeping — only its
 * identity. An application session adds whatever per-session state it needs ON TOP (the
 * default {@link Session} adds a `data` bag); MCP's {@link
 * import('../mcp/MCPSession.js').MCPSession} adds the resumable replay log. `createSession`
 * is generic over `S extends SessionInterface`, so the resolved session set on
 * `context.state` is the consumer's concrete type, narrowed (never asserted, §14).
 */
export interface SessionInterface {
	readonly id: string
}

/**
 * The pluggable storage seam behind {@link import('./middlewares.js').createSession} — the
 * smallest set of primitives the middleware needs to resolve, persist, and end a session
 * (AGENTS §21). The session analogue of {@link import('@src/core').QueueStoreInterface}, but
 * LEANER: just `get` / `set` / `delete` (the middleware never enumerates or bulk-clears, so
 * there is no `load` / `clear`).
 *
 * @remarks
 * The store OWNS the idle-TTL policy: `get` applies the expiry sweep and the touch-on-access
 * internally, so the session value stays pure identity (it carries no `lastSeen` — the store
 * tracks it). `createSession` defaults to an in-memory {@link
 * import('./stores/MemorySessionStore.js').MemorySessionStore} (the lifted closure store), and a
 * durable / DB-backed store can REPLACE it via the `store` option without touching the
 * middleware — exactly as a {@link import('@src/core').QueueStoreInterface} backs a queue.
 * Every primitive is async (a `Promise`), so a durable backend (a database round-trip) fits
 * the same shape as the memory one. The session carries its OWN id, so `set` takes no
 * separate id param (mirroring the queue store's `save`, which keys off `entry.id`).
 *
 * @typeParam S - The concrete session type the store holds (defaults to {@link Session})
 */
export interface SessionStoreInterface<S extends SessionInterface = Session> {
	/**
	 * Resolve a LIVE session by id — the store applies idle-TTL expiry + touches `lastSeen`
	 * internally; an absent or expired id resolves `undefined`.
	 *
	 * @param id - The session id to resolve
	 * @param now - The clock instant (epoch ms) the TTL sweep evaluates against; defaults to
	 *   `Date.now()`. The deterministic clock seam — {@link
	 *   import('./middlewares.js').createSession} threads its `clock()` here, so an injected
	 *   clock drives eviction with no wall-clock race (AGENTS §16)
	 * @returns The live session, or `undefined` if absent / expired
	 */
	get(id: string, now?: number): Promise<S | undefined>
	/**
	 * Insert or replace a session under its own `session.id` (no separate id param —
	 * mirroring {@link QueueStoreInterface.save}), stamping `lastSeen = now`.
	 *
	 * @param session - The session to store (keyed by its `id`)
	 * @param now - The clock instant (epoch ms) `lastSeen` (and a first-set `createdAt`) is
	 *   stamped with; defaults to `Date.now()` — the same clock seam `get` carries
	 */
	set(session: S, now?: number): Promise<void>
	/**
	 * Drop a session by id (logout / teardown); an absent id is a no-op (no throw).
	 *
	 * @param id - The session id to drop
	 */
	delete(id: string): Promise<void>
}

/**
 * Options for {@link import('./factories.js').createMemorySessionStore} — the in-memory
 * {@link SessionStoreInterface}.
 *
 * @remarks
 * - `ttl` — the session IDLE lifetime in milliseconds: a session not accessed within `ttl`
 *   is lazily evicted on the next `get` (read as absent + dropped — no background timer, the
 *   `createRateLimiter` lazy-window idiom). Every live `get` TOUCHES the session, so an actively-used
 *   one never idle-ages-out. Omit it for sessions that live until an explicit `delete`.
 * - `absoluteTtl` — the session ABSOLUTE lifetime (ms) from MINT, INDEPENDENT of the idle `ttl`
 *   touch: a session older than `absoluteTtl` is evicted on the next `get` (read as absent) EVEN IF
 *   continuously active. It caps the total lifetime of a hijacked / long-lived session so a stolen
 *   id cannot be kept alive forever by simply being used — the `createdAt` is stamped at mint (the
 *   first `set` for an id) and PRESERVED across the way-out re-persists, so activity never resets it.
 *   Omit it for sessions bounded only by idle `ttl`.
 */
export interface MemorySessionStoreOptions {
	readonly ttl?: number
	readonly absoluteTtl?: number
}

/**
 * The row a {@link import('./stores/DatabaseSessionStore.js').DatabaseSessionStore} persists — one
 * session per id, the session value held as ONE OPAQUE JSON column plus the store-owned idle-TTL
 * cursor.
 *
 * @remarks
 * `id` mirrors the session's own {@link SessionInterface.id} (the primary key). `session` is the
 * session value as one opaque JSON blob (a `rawShape` column the factory builds) — read back as the
 * broad `unknown` and narrowed to the concrete session on `get` ({@link
 * import('./helpers.js').isSession}, the AGENTS §14 boundary narrow), exactly as {@link
 * import('@src/core').DatabaseQueueStore} stores its `input`: the open generic `S` keeps the row
 * type FLAT, so a typed-`S` column never forces a TS2589-tripping inference. **`lastSeen` is the
 * `number` (epoch ms) instant the session was last accessed** — the idle-TTL cursor the STORE owns
 * (so {@link SessionInterface} stays pure identity), mirroring the memory twin's per-entry `lastSeen`.
 * **`createdAt` is the `number` (epoch ms) MINT instant** — the absolute-TTL cursor, stamped at the
 * first `set` for an id and PRESERVED across the way-out re-persists so activity never resets it.
 */
export interface SessionRow {
	readonly id: string
	/** The whole session value as one opaque JSON blob — read back as `unknown`, narrowed on `get`. */
	readonly session: unknown
	/** The epoch-ms instant the session was last accessed — the store-owned idle-TTL cursor. */
	readonly lastSeen: number
	/** The epoch-ms instant the session was minted — the store-owned absolute-TTL cursor (set once). */
	readonly createdAt: number
}

/**
 * Options for `createSession` — the generic, reusable session {@link Middleware}.
 *
 * @remarks
 * Every field is optional; the zero-arg `createSession()` is an AUTO-SESSION (mint a fresh
 * {@link Session} whenever a request carries no valid one, proceed otherwise). The knobs
 * tune the four decisions the middleware makes — how to identify a session, how long it
 * lives, what to store, whether to mint, and what to do when one is required but absent.
 *
 * - `header` — the request/response header the session id travels in; defaults to {@link
 *   import('./constants.js').DEFAULT_SESSION_HEADER} (`'session-id'`). The middleware READS
 *   it to resolve an existing session and SETS it (on the response) on the mint that opens
 *   a new one. MCP overrides it to `'mcp-session-id'`.
 * - `ttl` — the session IDLE lifetime in milliseconds: a session not accessed within `ttl`
 *   reads as ABSENT and is lazily evicted on the next access (no background timer — the
 *   `createRateLimiter` lazy-window idiom). Omit it for sessions that live until an explicit
 *   `DELETE`.
 * - `absoluteTtl` — the session ABSOLUTE lifetime (ms) from mint, INDEPENDENT of the idle `ttl`
 *   touch: a session older than `absoluteTtl` is evicted on the next access EVEN IF continuously
 *   used, capping the total lifetime of a hijacked / long-lived session. Like `ttl`, it is consumed
 *   by the DEFAULT store and ignored once a custom `store` is supplied (the store owns the sweep).
 *   Omit it for sessions bounded only by idle `ttl`.
 * - `key` — the `context.state` key the resolved session is stashed under (for a downstream
 *   middleware / the handler to read); defaults to {@link
 *   import('./constants.js').SESSION_STATE}. MCP uses its own {@link
 *   import('../mcp/constants.js').MCP_SESSION_STATE}.
 * - `create` — how to build a fresh session from a minted id; defaults to a new {@link
 *   Session}. A consumer returns its own `S` (MCP returns an `MCPSession`).
 * - `mint` — whether to MINT a brand-new session for THIS request when it carries no valid
 *   one; defaults to ALWAYS mint (the auto-session). MCP returns `true` only for an
 *   `initialize` POST, so a bare non-initialize request is rejected rather than silently
 *   given a fresh session. May be async (it can read `context.body()`, which is cached so a
 *   downstream route re-reads it).
 * - `require` — when NOT minting and there is no valid session, whether to REJECT (`true`)
 *   or just proceed without one (`false`, the default — a best-effort auto-session that
 *   tolerates an absent id). MCP sets it `true` so a non-initialize request without a valid
 *   session is a hard `404`.
 * - `onMissing` — how to reject when `require` is set and no valid session resolved;
 *   defaults to a plain `404` (`context.empty(404)`). MCP passes its JSON-RPC error body.
 * - `store` — the {@link SessionStoreInterface} backing the per-id session storage; defaults
 *   to an in-memory {@link import('./stores/MemorySessionStore.js').MemorySessionStore} (built from
 *   `ttl`). Supply a durable / DB-backed store to persist sessions across restarts (or to
 *   share one store across several middlewares) — the store OWNS the idle-TTL sweep, so `ttl`
 *   is consumed by the DEFAULT store and ignored once a custom `store` is given.
 * - `transport` — the {@link SessionTransport} that READS the id from / WRITES it to the
 *   request/response; defaults to the HEADER transport (the `header` knob). {@link
 *   import('./middlewares.js').createCookieSession} swaps in a signed-cookie transport so the
 *   id rides a `Set-Cookie` / `Cookie` instead of a header, REUSING this whole machine — every
 *   other knob (store, mint, require, lifecycle) is unchanged. When a custom `transport` is
 *   given, the `header` knob is the transport's concern (the header transport reads it; the
 *   cookie transport ignores it).
 * - `clock` — the `() => number` epoch-ms clock the middleware threads into every store
 *   `get` / `set` it makes (the {@link SessionStoreInterface} trailing-`now` seam); defaults
 *   to `Date.now`. The deterministic clock seam: a test injects a manual clock and drives
 *   TTL eviction by advancing it explicitly — never racing a real TTL window against
 *   wall-clock (AGENTS §16). Production never sets it.
 */
export interface SessionOptions<S extends SessionInterface = Session> {
	readonly header?: string
	readonly ttl?: number
	readonly absoluteTtl?: number
	readonly key?: string
	readonly create?: (id: string) => S
	readonly mint?: (context: RouteHandlerContextInterface) => boolean | Promise<boolean>
	readonly require?: boolean
	readonly onMissing?: (context: RouteHandlerContextInterface) => void
	readonly store?: SessionStoreInterface<S>
	readonly transport?: SessionTransport
	readonly clock?: () => number
	readonly deleteEnds?: boolean
}

/**
 * The id TRANSPORT seam behind {@link import('./middlewares.js').createSession} — how the
 * session id is READ from the request, WRITTEN to the response on a mint / rotation, and
 * CLEARED on a `destroy`. It factors the wire location out of the session lifecycle so the
 * SAME resolve / mint / validate / TTL machine drives a HEADER session ({@link createSession}'s
 * default) or a signed-COOKIE one ({@link import('./middlewares.js').createCookieSession}) — the
 * minimal-interface seam (AGENTS §21), the transport analogue of the pluggable {@link
 * SessionStoreInterface}.
 *
 * @remarks
 * The DEFAULT transport reads/writes the configured `header`; the cookie transport signs the id
 * into a `Set-Cookie` and reads it back from the `Cookie` header. `read` is TOTAL — it narrows
 * the untrusted request input (a raw header, a cookie string) with a guard / `typeof`, never an
 * assertion (AGENTS §14), and yields `undefined` for an absent / malformed / tampered id.
 * `write` sets the id on the response (the mint / rotation path); `clear` removes it (the
 * `destroy` path — for a cookie that is a `Max-Age=0` expiry, for a header a no-op).
 */
export interface SessionTransport {
	/**
	 * Read the session id off the request — TOTAL, narrowing untrusted input with a guard,
	 * never an assertion (§14); an absent / malformed / tampered id yields `undefined`.
	 *
	 * @param context - The request's {@link RouteHandlerContextInterface}
	 * @returns The resolved session id, or `undefined` when none is present
	 */
	read(context: RouteHandlerContextInterface): string | undefined
	/**
	 * Write the session id to the response — the mint / rotation path (a header set, or a
	 * signed `Set-Cookie`).
	 *
	 * @param context - The request's {@link RouteHandlerContextInterface}
	 * @param id - The session id to emit
	 */
	write(context: RouteHandlerContextInterface, id: string): void
	/**
	 * Clear the session id from the response — the `destroy` path (a `Max-Age=0` cookie
	 * expiry; a no-op for a header transport that has nothing to revoke).
	 *
	 * @param context - The request's {@link RouteHandlerContextInterface}
	 */
	clear(context: RouteHandlerContextInterface): void
}

/**
 * The per-request session CONTROL handle {@link import('./middlewares.js').createSession} stashes
 * on `context.state` (under {@link import('./constants.js').SESSION_CONTROL_STATE}) alongside the
 * resolved session — the two lifecycle niceties a handler invokes, which the middleware ACTS ON on
 * the way out (after `next()`), keeping the {@link Session} entity store-decoupled (AGENTS §21).
 *
 * @remarks
 * The handler FLAGS intent (`session.regenerate()` / `session.destroy()`); the middleware reads the
 * flag after `next()` and performs the storage + transport work — so the session value itself never
 * touches the store or the wire. Both are idempotent flags (calling twice is the same as once); the
 * last one set on the way out wins (`destroy` after `regenerate` ends the session). Reached via
 * `context.state.get(SESSION_CONTROL_STATE)`, narrowed with {@link
 * import('./helpers.js').isSessionControl} (never asserted, §14).
 *
 * - **`regenerate()`** — rotate the id (anti-fixation): on the way out the middleware MINTS a fresh
 *   id, `store.set`s the SAME session data under it, `store.delete`s the old id, and WRITES the new
 *   id via the transport. The session DATA is preserved; only the id changes.
 * - **`destroy()`** — end the session (logout): on the way out the middleware `store.delete`s it and
 *   CLEARS the id via the transport (a `Max-Age=0` cookie). Nothing is persisted.
 */
export interface SessionControlInterface {
	/** Flag the session for id rotation on the way out (anti-fixation) — data preserved, id changes. */
	regenerate(): void
	/** Flag the session to END on the way out (logout) — `store.delete` + transport clear. */
	destroy(): void
}

/**
 * The `Set-Cookie` attributes for {@link import('./helpers.js').serializeCookie} (and the cookie
 * the signed-cookie transport / {@link import('./middlewares.js').createCSRFGuard} emit).
 *
 * @remarks
 * Each leaf is a single word (AGENTS §4.1); every field is optional and omitted by default except
 * where noted. The attribute names map to the `Set-Cookie` directives one-for-one.
 *
 * - `path` — the `Path` directive; defaults to `'/'` (the cookie applies to the whole origin).
 * - `domain` — the `Domain` directive; omitted by default (host-only cookie).
 * - `maxAge` — the `Max-Age` directive in SECONDS (the cookie wire unit, NOT the ms `ttl`); a `0`
 *   expires the cookie immediately (the `destroy` / clear path). Omitted ⇒ a session cookie.
 * - `httpOnly` — the `HttpOnly` directive (hidden from client JS); defaults to `true`. The CSRF
 *   token cookie sets it `false` so the double-submit token is READABLE by client JS to echo.
 * - `secure` — the `Secure` directive (HTTPS-only). The session / CSRF cookie writers default it to
 *   `'auto'` — `Secure` when the request arrived over TLS (`request.socket.encrypted === true`), OFF
 *   over plaintext HTTP — so a direct-HTTPS deployment is `Secure` automatically while local-dev HTTP
 *   still works, with no per-environment config. Set it `true` to FORCE `Secure` (behind a TLS-
 *   terminating proxy that re-presents the request as plaintext HTTP, where `'auto'` would read no
 *   TLS), or `false` to force it off. NOTE: a `sameSite: 'None'` cookie is ALWAYS `Secure` regardless
 *   (the spec / browsers require it). The bare {@link import('./helpers.js').serializeCookie} builder
 *   treats `'auto'` (no request in scope) as OFF; the request-aware writers resolve it first via
 *   {@link import('./helpers.js').resolveSecure}.
 * - `sameSite` — the `SameSite` directive; defaults to `'Lax'`. `'Strict'` for the CSRF cookie. A
 *   `'None'` cookie FORCES `Secure` (spec requirement) — {@link import('./helpers.js').serializeCookie}
 *   adds it even when `secure` is unset / `false`.
 */
export interface CookieOptions {
	readonly path?: string
	readonly domain?: string
	readonly maxAge?: number
	readonly httpOnly?: boolean
	readonly secure?: boolean | 'auto'
	readonly sameSite?: 'Strict' | 'Lax' | 'None'
}

 * @remarks

 * inherits every {@link SessionOptions} knob EXCEPT `header` / `transport` (which the cookie
 * transport supplies). The cookie-specific knobs are added:
 *
 * - `secret` — the {@link TokenSecret} the id cookie is HMAC-signed with (REQUIRED; a rotation

/**
 * Options for {@link import('./middlewares.js').createCSRFGuard} — the double-submit signed-cookie
 * CSRF {@link Middleware}.
 *
 * @remarks
 * The classic stateless double-submit pattern: on a SAFE method the guard mints a token, sets it as
 * a SIGNED cookie (`SameSite=Strict`, `HttpOnly:false` so client JS can read it, `Secure` AUTO — on
 * over TLS, off over plaintext HTTP), and exposes the RAW token on `context.state` (under `key`) for
 * the handler to echo into its HTML / a response header. On a MUTATING method it reads the submitted
 * token (from `header`, else the body `field`), verifies the cookie's signature, and rejects
 * (`HTTPError(403)`) on a missing / invalid / mismatched token — else continues. It needs NO
 * server-side store (the signed cookie IS the server state).
 *
 * **Session-binding.** When a session is present on `context.state` (under `session`, the key the
 * session middleware set), the minted token is BOUND to the session id (`signToken(sessionId)`) and a
 * mutating request must present the token bound to ITS OWN session — so a cookie-injection attacker
 * (a sibling subdomain / MITM) who plants both double-submit halves under another user's session is
 * still rejected. Register the guard AFTER `createSession` / `createCookieSession` to activate it;
 * sessionless, it falls back to a signed-random token (the plain double-submit, weaker).
 *
 * - `secret` — the {@link TokenSecret} the token cookie is HMAC-signed with (REQUIRED; a rotation
 *   list accepts a cookie signed by any of its secrets).
 * - `cookie` — the token cookie NAME; defaults to {@link import('./constants.js').DEFAULT_CSRF_COOKIE}
 *   (`'csrf'`).
 * - `header` — the request header the submitted token is read from on a mutating method; defaults
 *   to {@link import('./constants.js').DEFAULT_CSRF_HEADER} (`'x-csrf-token'`).
 * - `field` — the parsed-body field the submitted token is read from when the `header` is absent
 *   (an HTML-form POST); defaults to {@link import('./constants.js').DEFAULT_CSRF_FIELD} (`'_csrf'`).
 *   Reading it requires the body already parsed (mount {@link createBodyParser} ahead of the guard).
 * - `key` — the `context.state` key the raw token is exposed under on a safe method; defaults to
 *   {@link import('./constants.js').CSRF_STATE} (`'csrf'`).
 * - `session` — the `context.state` key the guard reads the CURRENT session from to BIND the token to
 *   the session id; defaults to {@link import('./constants.js').SESSION_STATE} (`'session'`), the
 *   key `createSession` / `createCookieSession` set. Override it only when the paired session
 *   middleware uses a custom `key`. When no session resolves under it, the guard is plain double-submit.
 * - `safe` — the methods that SKIP verification + mint a token; defaults to {@link
 *   import('./constants.js').DEFAULT_CSRF_SAFE} (`GET` / `HEAD` / `OPTIONS`).
 */
export interface CSRFOptions {
	readonly secret: TokenSecret
	readonly cookie?: string
	readonly header?: string
	readonly field?: string
	readonly key?: string
	readonly session?: string
	readonly safe?: readonly HTTPMethod[]
}

/**
 * Options for {@link import('./middlewares.js').createSecurity} — the security-header {@link
 * Middleware}.
 *
 * @remarks
 * Every header is opt-out (a secure default ON), so the zero-arg `createSecurity()` sets the full
 * recommended set. A header with a configurable VALUE takes a `string | false` union — a `string`
 * OVERRIDES the default value, `false` SUPPRESSES the header (the §4.4 "value, or off" union, not a
 * behavioral toggle); a value-less header (`frame` ⇒ the `X-Frame-Options` token) takes its own
 * union too. `X-Content-Type-Options: nosniff` is unconditional (it has no downside and no value to
 * tune), so it is not an option.
 *
 * - `frame` — the `X-Frame-Options` value (clickjacking defense): `'DENY'` (the default) or
 *   `'SAMEORIGIN'`, or `false` to omit the header.
 * - `hsts` — the `Strict-Transport-Security` value; OFF by default (`false`) because it is only
 *   meaningful — and is browser-pinned, so a wrong value is sticky — over HTTPS. `true` sets the
 *   recommended {@link import('./constants.js').DEFAULT_HSTS} directive; a `string` overrides it.
 * - `csp` — the `Content-Security-Policy` value; defaults to {@link
 *   import('./constants.js').DEFAULT_CSP} (`default-src 'self'` PLUS `base-uri 'self'` / `object-src
 *   'none'` / `frame-ancestors 'self'` / `form-action 'self'` — the injection + clickjacking
 *   hardening directives). A `string` REPLACES the default WHOLESALE (it is not merged), `false` omits
 *   the header.
 * - `referrer` — the `Referrer-Policy` value; defaults to {@link
 *   import('./constants.js').DEFAULT_REFERRER_POLICY}, a `string` overrides it, `false` omits it.
 * - `permissions` — the `Permissions-Policy` value; defaults to {@link
 *   import('./constants.js').DEFAULT_PERMISSIONS_POLICY}, a `string` overrides it, `false` omits it.
 * - `coop` — the `Cross-Origin-Opener-Policy` value; defaults to {@link
 *   import('./constants.js').DEFAULT_COOP} (`'same-origin'`, process-isolating the browsing context —
 *   the Spectre / XS-Leaks defense that severs cross-window references). A `string` overrides it (e.g.
 *   `'same-origin-allow-popups'`), `false` omits it.
 * - `corp` — the `Cross-Origin-Resource-Policy` value; defaults to {@link
 *   import('./constants.js').DEFAULT_CORP} (`'same-origin'`, blocking another origin from embedding the
 *   response). A `string` overrides it (`'cross-origin'` for a public CDN asset), `false` omits it.
 * - `coep` — the `Cross-Origin-Embedder-Policy` opt-IN (a BOOLEAN, OFF by default): `true` sets {@link
 *   import('./constants.js').DEFAULT_COEP} (`'require-corp'`) to enable cross-origin ISOLATION; OFF by
 *   default because `require-corp` BREAKS cross-origin subresources for most apps, so — unlike the
 *   value-bearing headers — it is opt-in, never part of the secure-default set. A `string` overrides the
 *   emitted directive when enabled.
 * - `requestId` — the request-correlation id (§4.2.1 entity-keyed group): when ENABLED (the default),
 *   the middleware sets an `X-Request-ID` response header AND stashes the id on `context.state` under
 *   {@link import('./constants.js').REQUEST_ID_STATE} for downstream logging / correlation. `trust`
 *   (default `false`) honors an INCOMING `X-Request-ID` — but ONLY a client id matching a STRICT charset
 *   ({@link import('./helpers.js').isValidRequestId}, `^[A-Za-z0-9_-]{1,200}$`); an absent / malformed /
 *   over-long / control-char-bearing incoming id is REJECTED and a fresh `crypto.randomUUID()` is minted
 *   instead, so a CRLF / log-injection / oversized header NEVER rides into the response header or
 *   `context.state` (§14). With `trust` `false` (the default) the incoming header is ignored entirely and
 *   a UUID is always minted. `false` disables the id entirely.
 */
export interface SecurityOptions {
	readonly frame?: 'DENY' | 'SAMEORIGIN' | false
	readonly hsts?: string | boolean
	readonly csp?: string | false
	readonly referrer?: string | false
	readonly permissions?: string | false
	readonly coop?: string | false
	readonly corp?: string | false
	readonly coep?: string | boolean
	readonly requestId?: { readonly trust?: boolean } | false
}

/**
 * Options for {@link import('./middlewares.js').createETag} — the conditional-GET (ETag) {@link
 * Middleware} for DYNAMIC handler output.
 *
 * @remarks
 * Both fields are optional; the zero-arg `createETag()` adds a WEAK, SHA-256 content `ETag` to every
 * cacheable response (a `GET` `200`) and answers `304 Not Modified` on a matching
 * `If-None-Match`. It REUSES {@link import('./helpers.js').bufferResponse}, so it composes with {@link
 * import('./middlewares.js').createCompression} (mount ETag INNER) — the ETag is computed over the
 * UNCOMPRESSED representation, then compression compresses, and a conditional revalidation still
 * `304`s. Streaming / SSE responses are passed through untouched (an unbounded stream cannot be
 * buffered or hashed).
 *
 * - `weak` — emit a WEAK validator (`W/"<hash>"`, the default — encoding-agnostic, so it survives a
 *   downstream content re-encoding like gzip and still matches on revalidation). Set `false` for a
 *   STRONG validator (`"<hash>"`, byte-identity — only correct when nothing downstream re-encodes the
 *   body). The §4.4 boolean toggle (weak vs. strong).
 * - `algorithm` — the `node:crypto` hash the content `ETag` is computed with; defaults to {@link
 *   import('./constants.js').DEFAULT_ETAG_ALGORITHM} (`'sha256'`).
 */
export interface ETagOptions {
	readonly weak?: boolean
	readonly algorithm?: string
}

/**
 * The parsed outcome of an HTTP `Range` request header — the result of {@link
 * import('./helpers.js').parseRange}.
 *
 * @remarks
 * A `Range: bytes=start-end` against a known resource `size` resolves to ONE of two shapes,
 * discriminated by `satisfiable` (the axis is whether the requested span overlaps the resource):
 *
 * - **`satisfiable: true`** — a concrete, clamped byte window `[start, end]` (INCLUSIVE, the HTTP
 *   wire convention), normalized from the header's open / suffix / closed forms against `size`. The
 *   middleware answers a `206` with a `Content-Range: bytes start-end/size` and streams exactly that
 *   slice (`fs.createReadStream(path, { start, end })`).
 * - **`satisfiable: false`** — the range lies wholly outside the resource (an unsatisfiable range).
 *   The middleware answers a `416` whose `Content-Range` is the unsatisfiable form (`bytes` then an
 *   asterisk, a slash, and the total size) and no body.
 *
 * {@link import('./helpers.js').parseRange} returns `undefined` for an ABSENT / unparseable / multi-
 * range / non-`bytes` header — the "no range, serve the whole resource as a `200`" case — so the
 * three outcomes (full / partial / unsatisfiable) are distinguished without a separate flag. It is
 * TOTAL — a hostile header never throws, it yields `undefined` (AGENTS §14).
 */
export type RangeSpec =
	| { readonly satisfiable: true; readonly start: number; readonly end: number }
	| { readonly satisfiable: false }

/**
 * Options for {@link import('./middlewares.js').createStatic} — the static-file-serving {@link
 * Middleware} (server-only `node:fs`).
 *
 * @remarks
 * Serves a directory of files over HTTP, composably: a request that does not resolve to a file under
 * `root` falls through via `next()` (so other routes still run), unless the SPA `fallback` claims it.
 * Only `root` is required; the rest tune resolution, caching, and the client-router fallback. The
 * highest-risk piece is the TRAVERSAL GUARD — the resolved absolute path is REQUIRED to stay within
 * the resolved `root`, so a `../`, an encoded `..`, an absolute path, or a null byte never reads
 * outside `root` (it is treated as not-found, never served). Every untrusted input (the decoded
 * pathname, the `Range` / `If-None-Match` / `Accept` headers) is narrowed, never asserted (AGENTS §14).
 *
 * - `root` — the absolute directory to serve files from (REQUIRED). The traversal guard resolves
 *   every request path UNDER this root and refuses to serve anything outside it.
 * - `prefix` — an optional URL path prefix stripped before resolving (e.g. `'/assets'` serves
 *   `root/app.js` for `GET /assets/app.js`). A request whose pathname does not start with `prefix`
 *   falls through (`next()`). Omitted ⇒ the whole pathname maps under `root`.
 * - `index` — the file served for a directory request (a pathname resolving to a directory under
 *   `root`); defaults to {@link import('./constants.js').DEFAULT_STATIC_INDEX} (`'index.html'`). The
 *   SPA `fallback` serves this same document. An absent index ⇒ the directory falls through.
 * - `dotfiles` — the policy for a path with a segment starting `.` (a dotfile / dot-directory):
 *   `'ignore'` (the default {@link import('./constants.js').DEFAULT_DOTFILES}) treats it as not-found
 *   (falls through), `'deny'` answers a `403`, `'allow'` serves it. Not a behavioral toggle (three
 *   distinct policies), so it stays a union (AGENTS §4.4).
 * - `maxAge` — the `Cache-Control: max-age=<maxAge>` freshness lifetime in SECONDS sent with a served
 *   file; omitted ⇒ no `Cache-Control` header.
 * - `etag` — whether to send a WEAK `ETag` (`W/"<size>-<mtimeMs>"`) and honor a conditional `304` on
 *   a matching `If-None-Match`; defaults to `true` (a binary behavioral toggle, so a boolean — §4.4).
 * - `fallback` — the SPA document fallback: when enabled, a GET request that did NOT resolve to a file
 *   AND is extension-less AND accepts `text/html` AND is not under `exclude` is served the `index`
 *   document (so a client-side router owns the route). `true` enables it with the default exclude
 *   ({@link import('./constants.js').DEFAULT_STATIC_FALLBACK_EXCLUDE}, an `/api` prefix); an object
 *   overrides `exclude`; omitted / `false` ⇒ off (a missing file falls through as a `next()`).
 */
export interface StaticOptions {
	readonly root: string
	readonly prefix?: string
	readonly index?: string
	readonly dotfiles?: 'ignore' | 'deny' | 'allow'
	readonly maxAge?: number
	readonly etag?: boolean
	readonly fallback?: boolean | { readonly exclude?: string }
}

/**
 * A content-coding the spine compresses / decompresses with — the `Content-Encoding` /
 * `Accept-Encoding` token vocabulary it understands.
 *
 * @remarks
 * `gzip` / `deflate` / `br` (Brotli) are the three real codings (mapped to `node:zlib`'s
 * `gzip` / `deflate` / `brotliCompress`); `identity` is the no-op "send it uncompressed"
 * coding. It is a constrained set of external-spec literals (the IANA content-codings), so
 * it stays a union, not a behavioral toggle (AGENTS §4.4). The acronym `br` keeps its
 * canonical lowercase wire spelling.
 */
export type Encoding = 'gzip' | 'deflate' | 'br' | 'identity'

/**
 * One parsed entry of a weighted `Accept` / `Accept-Encoding` / `Accept-Language` header —
 * a media type / coding / language `value` and its quality `q` — the element type {@link
 * import('./helpers.js').parseAcceptHeader} returns (sorted by `q` descending).
 *
 * @remarks
 * - `value` — the lower-cased token (`text/html`, `gzip`, `en-us`, or a wildcard `*` / `* /*`).
 * - `q` — the quality weight in `[0, 1]` (the `;q=` parameter, default `1` when absent). A
 *   `;q=0` entry explicitly REJECTS that token — a parser keeps it (so a caller can honor the
 *   rejection) rather than dropping it.
 */
export interface AcceptEntry {
	readonly value: string
	readonly q: number
}

/**
 * Options for {@link import('./middlewares.js').createCompression} — the response-compression
 * {@link Middleware}.
 *
 * @remarks
 * Both fields are optional; the zero-arg `createCompression()` compresses any buffered response
 * at or above the default threshold whose `Content-Type` is compressible, negotiating the best
 * coding from the request's `Accept-Encoding`.
 *
 * - `threshold` — the minimum buffered body size in BYTES before compression is worthwhile; a
 *   body smaller than this is sent uncompressed (compressing a tiny payload costs more than it
 *   saves, and can grow it). Defaults to {@link import('./constants.js').DEFAULT_COMPRESSION_THRESHOLD}.
 * - `encodings` — the {@link Encoding} codings the middleware is willing to use, in PREFERENCE
 *   order (the first acceptable one whose `q` is highest wins ties). Defaults to {@link
 *   import('./constants.js').DEFAULT_COMPRESSION_ENCODINGS} (`br` / `gzip` / `deflate`). `identity`
 *   is always implicitly acceptable (an un-listed / `identity`-only client gets the body uncompressed).
 */
export interface CompressionOptions {
	readonly threshold?: number
	readonly encodings?: readonly Encoding[]
}

/**
 * A handle that BUFFERS a {@link ServerResponse}'s output instead of writing it to the socket — the
 * reusable response-buffering machine {@link import('./helpers.js').bufferResponse} installs, behind
 * {@link import('./middlewares.js').createCompression} (and any future middleware that must see the
 * whole body before it is sent, e.g. an ETag middleware).
 *
 * @remarks
 * It wraps `writeHead` / `write` / `end` / `flushHeaders` so a complete (non-streaming) response is
 * captured rather than flushed — the consumer then transforms it (compress, hash) and calls `send`
 * to write the final bytes. It auto-bypasses to PASS-THROUGH the instant the handler STREAMS (an
 * SSE `text/event-stream` head, an explicit `flushHeaders()`, or a raw `write()` before `end()` —
 * the response senders never call `write()`), because an unbounded stream cannot be buffered: on
 * bypass it restores the real methods and replays the captured head + buffered chunks straight to
 * the socket, and `streaming` reads `true` thereafter.
 *
 * - `streaming` — `true` once the buffer bypassed to pass-through (the handler streamed). When `true`
 *   the response is already being written directly; the consumer does NOTHING (no `send`).
 * - `ended` — `true` once the buffered response logically ended (a buffered `end()` ran) WITHOUT
 *   streaming — i.e. a complete body is captured and ready for the consumer to transform + `send`.
 * - `status` — the captured response status code (the `writeHead` status, or the live `statusCode`).
 * - `headers()` — the captured response headers, lower-cased keys, merging `setHeader` + `writeHead`.
 *   A single-valued header is a string; a MULTI-valued one (notably `set-cookie`, which HTTP forbids
 *   comma-folding) stays a `string[]`, so two cookies survive the buffer intact (the
 *   {@link import('./helpers.js').collectHeaders} normalization).
 * - `body()` — the concatenated buffered body as one {@link Buffer} (empty for a body-less response).
 * - `send(headers, body, status?)` — restore the real response methods and write the final response:
 *   `writeHead(status ?? this.status, headers)` then `end(body)`. The consumer passes its transformed
 *   headers (e.g. a new `Content-Encoding` + `Content-Length`) and body; `status` overrides only when
 *   needed (e.g. a `304`). A no-op once the head is genuinely sent.
 * - `release()` — RESTORE the real response methods and replay any captured head + chunks straight to
 *   the socket WITHOUT taking over the write, then let the response proceed normally. The consumer
 *   calls it when it will NOT `send` and the response did not stream + did not end (an INCOMPLETE
 *   buffered response — e.g. an inner handler threw before ending, and an OUTER error boundary will
 *   write the response itself). It unwedges the response so that outer write lands on the real socket.
 *   Idempotent + a no-op once streaming / sent.
 */
export interface ResponseBufferInterface {
	readonly streaming: boolean
	readonly ended: boolean
	readonly status: number
	headers(): Record<string, string | string[]>
	body(): Buffer
	send(headers: Record<string, string | string[]>, body: Buffer, status?: number): void
	release(): void
}

/**
 * A map of media type → handler for {@link NegotiatorInterface.format} — the content-negotiation
 * dispatch table.
 *
 * @remarks
 * Each key is a media type the route can PRODUCE (`application/json`, `text/html`, …) and each
 * value the {@link RouteHandler}-shaped responder that writes that representation. `format`
 * negotiates the client's preferred key from the request `Accept` header and invokes the matching
 * handler, or answers `406 Not Acceptable` when none of the offered types is acceptable. The keys
 * are the `available` set negotiation chooses from, so their order is the server's tie-break
 * preference (a client accepting the any-media-range gets the first key).
 */
export type FormatHandlerMap = Readonly<
	Record<string, (context: RouteHandlerContextInterface) => void | Promise<void>>
>

/**
 * Content negotiation over the weighted `Accept` family — a reusable, cross-middleware machine
 * (NOT a middleware), constructed via {@link import('./factories.js').createNegotiator}.
 *
 * @remarks
 * It parses a weighted `Accept` / `Accept-Encoding` / `Accept-Language` header (q-values honored,
 * `*` wildcards honored, a `;q=0` rejection respected) and picks the client's most-preferred value
 * that the server also offers. `negotiate` is the generic primitive (media types); `encoding` /
 * `language` are its sibling axes (a coding wildcard `*`, a language prefix match `en` ⇒ `en-US`).
 * `format` is the dispatcher — it reads the request `Accept`, negotiates against the handler map's
 * media types, and invokes the winning handler, or sends `406` when none is acceptable. It is
 * SHARED machinery (the §5 distinction): {@link import('./middlewares.js').createCompression} reuses
 * the SAME q-value parser ({@link import('./helpers.js').parseAcceptHeader}) for its `Accept-Encoding`
 * negotiation, so there is one parser, not two.
 */
export interface NegotiatorInterface {
	/**
	 * Pick the best `available` value for a weighted `Accept`-style `header` — the generic
	 * media-type primitive (`encoding` / `language` build on it).
	 *
	 * @remarks
	 * Parses `header` into q-sorted {@link AcceptEntry}s and returns the first `available` value the
	 * client accepts: an exact match, a subtype wildcard like `text/` then `*` (matched on the major
	 * type), or the any-media-range `*` then `/` then `*` (⇒ the first `available`). A `;q=0` entry
	 * never matches (an explicit rejection). Returns `undefined` when nothing is acceptable (or
	 * `available` is empty).
	 *
	 * @param header - The raw weighted header value (e.g. `text/html, application/json;q=0.9`)
	 * @param available - The values the server can produce, in preference (tie-break) order
	 * @returns The best acceptable value, or `undefined` when none is
	 */
	negotiate(header: string, available: readonly string[]): string | undefined
	/**
	 * Pick the best `available` content-coding for an `Accept-Encoding` header — `negotiate`
	 * scoped to codings (a bare `*` wildcard ⇒ the first `available`).
	 *
	 * @param header - The raw `Accept-Encoding` header value (e.g. `br;q=1.0, gzip;q=0.8`)
	 * @param available - The codings the server offers, in preference order
	 * @returns The best acceptable coding, or `undefined` when none is
	 */
	encoding(header: string, available: readonly string[]): string | undefined
	/**
	 * Pick the best `available` language for an `Accept-Language` header — `negotiate` with a
	 * language-prefix match (`en` accepts `en-US`) and a bare `*` wildcard.
	 *
	 * @param header - The raw `Accept-Language` header value (e.g. `en-US, en;q=0.8, fr;q=0.5`)
	 * @param available - The languages the server offers, in preference order
	 * @returns The best acceptable language, or `undefined` when none is
	 */
	language(header: string, available: readonly string[]): string | undefined
	/**
	 * Dispatch to the handler whose media type the client most prefers — read the request `Accept`,
	 * negotiate against `handlers`' keys, and invoke the winner; `406` when none is acceptable.
	 *
	 * @remarks
	 * An ABSENT `Accept` header is treated as the any-media-range (the client accepts anything → the
	 * first handler key). When no offered type is acceptable it sends a `406 Not Acceptable` via the
	 * context's `error` responder (so it composes with the error boundary). The chosen handler may be async.
	 *
	 * @param context - The request's {@link RouteHandlerContextInterface}
	 * @param handlers - The media type → responder {@link FormatHandlerMap} the route offers
	 */
	format(context: RouteHandlerContextInterface, handlers: FormatHandlerMap): Promise<void>
}

/**
 * The discriminant naming WHY a {@link import('./errors.js').MultipartError} was thrown — the axis it
 * splits on, never `kind` (AGENTS §4.4).
 *
 * @remarks
 * - `limit` — a per-request DoS bound tripped mid-stream (a file / field too large, too many files /
 *   fields, the total body too large) → a `413`.
 * - `malformed` — the body was not parseable as `multipart/form-data` (a missing / bad boundary, a
 *   truncated part, a header block that never closed) → a `400`.
 * - `rejected` — a file part failed the `allowed` MIME whitelist / magic-byte validation → a `415`.
 */
export type MultipartReason = 'limit' | 'malformed' | 'rejected'

/**
 * The lifecycle state of an {@link UploadedFileInterface} the multipart parser produced.
 *
 * @remarks
 * A constrained set of distinct states (not a behavioral toggle, so a union — AGENTS §4.4). A file
 * the handler receives in `context.body()` is always `staged` (a clean parse only ever hands back
 * staged files; a REJECTED file — a whitelist miss / magic-byte mismatch — or a limit breach aborts
 * the WHOLE parse with a {@link MultipartError} after cleaning every staged temp file, so the handler
 * never receives a bad file). `moved` is reached only by the handler calling
 * {@link import('./helpers.js').moveUploadedFile}.
 *
 * - `staged` — the part streamed in full to its temp file (and, when a whitelist applies, PASSED
 *   magic-byte validation); the app OWNS the temp file (it persists until the handler `move`s it or
 *   the app deletes it).
 * - `moved` — {@link import('./helpers.js').moveUploadedFile} relocated the temp file to its final
 *   destination (the returned record's `path` is now that destination; the original temp path no
 *   longer exists).
 */
export type UploadStatus = 'staged' | 'moved'

/**
 * One file part the multipart parser ({@link import('./middlewares.js').createMultipart}) streamed
 * out of a `multipart/form-data` request body and STAGED to a temp file — a FROZEN, immutable data
 * record (built by {@link import('./factories.js').createUploadedFile}); its operations are FREE
 * FUNCTIONS, never methods on the value.
 *
 * @remarks
 * Each file part (a `Content-Disposition` carrying a `filename`) is streamed to a temp file under a
 * RANDOM name (`crypto.randomUUID()`) in the OS temp directory — the client `name` (its original
 * filename) is untrusted METADATA and is NEVER used to build the path (a path-traversal defense: a
 * `filename="../../etc/passwd"` cannot escape the temp dir because the path is derived from a random
 * UUID, not the filename). The app OWNS the staged temp file after a successful parse: it reads the
 * bytes via {@link import('./helpers.js').streamUploadedFile} / {@link
 * import('./helpers.js').readUploadedFile} and takes ownership with {@link
 * import('./helpers.js').moveUploadedFile} (which returns a NEW frozen record), or deletes the file.
 *
 * - `field` — the form field name from the part's `Content-Disposition` (`name="avatar"`).
 * - `name` — the client's original filename (untrusted metadata; never the temp path).
 * - `size` — the total bytes staged.
 * - `mime` — the part's declared `Content-Type` (`image/png`), or `application/octet-stream` when the
 *   part omitted one.
 * - `validated` — whether the staged bytes' magic signature matched the declared `mime` against the
 *   {@link import('./constants.js').MAGIC_BYTES} table. With NO `allowed` whitelist this is recorded
 *   but never gates acceptance; with a whitelist, an un-`validated` file aborts the whole request
 *   (a {@link MultipartError} `415`), so a file the handler RECEIVES is always `validated: true`.
 * - `status` — the {@link UploadStatus} (`staged` after a clean parse, `moved` on the record {@link
 *   import('./helpers.js').moveUploadedFile} returns).
 * - `path` — the safe random temp path the bytes live at (inside the OS temp dir); on a moved record
 *   it is the destination path.
 */
export interface UploadedFileInterface {
	readonly field: string
	readonly name: string
	readonly size: number
	readonly mime: string
	readonly validated: boolean
	readonly status: UploadStatus
	readonly path: string
}

/**
 * The minimal data to create an {@link UploadedFileInterface} via {@link
 * import('./factories.js').createUploadedFile} — the §4.5 create bag (the streaming multipart parser
 * emits one per staged file part).
 *
 * @remarks
 * The same data members as {@link UploadedFileInterface}, except `status` is OPTIONAL — it defaults to
 * `'staged'` (a freshly staged file), so the parser need not spell it out. The free-function operations
 * ({@link import('./helpers.js').streamUploadedFile} / {@link import('./helpers.js').readUploadedFile} /
 * {@link import('./helpers.js').moveUploadedFile}) act on the FROZEN record the factory returns; `move`
 * sets `status: 'moved'` on the NEW record it returns.
 */
export interface UploadedFileInput {
	readonly field: string
	readonly name: string
	readonly size: number
	readonly mime: string
	readonly validated: boolean
	readonly status?: UploadStatus
	readonly path: string
}

/**
 * The per-request size / count limits {@link import('./middlewares.js').createMultipart} enforces
 * DURING the stream (the §4.2.1 entity-keyed group on {@link MultipartOptions}) — the DoS bound:
 * each check trips MID-STREAM and aborts early, so a hostile body is never buffered past the bound.
 *
 * @remarks
 * Every leaf is single-word and defaults to the matching `DEFAULT_MULTIPART_*` constant. Exceeding
 * ANY of them stops reading, CLEANS UP every already-staged temp file, and throws an
 * {@link HTTPError} `413`.
 *
 * - `file` — the maximum bytes a SINGLE file part may reach. Default
 *   {@link import('./constants.js').DEFAULT_MULTIPART_FILE}.
 * - `files` — the maximum NUMBER of file parts. Default
 *   {@link import('./constants.js').DEFAULT_MULTIPART_FILES}.
 * - `field` — the maximum bytes a SINGLE text field value may reach. Default
 *   {@link import('./constants.js').DEFAULT_MULTIPART_FIELD}.
 * - `fields` — the maximum NUMBER of text fields. Default
 *   {@link import('./constants.js').DEFAULT_MULTIPART_FIELDS}.
 * - `total` — the maximum TOTAL bytes across the whole request body (every part summed). Default
 *   {@link import('./constants.js').DEFAULT_MULTIPART_TOTAL}.
 */
export interface MultipartLimits {
	readonly file?: number
	readonly files?: number
	readonly field?: number
	readonly fields?: number
	readonly total?: number
}

/**
 * Options for {@link import('./middlewares.js').createMultipart} — the streaming
 * `multipart/form-data` parsing {@link Middleware} (server-only `node:fs` / `os` / `crypto`).
 *
 * @remarks
 * All fields are optional; the zero-arg `createMultipart()` parses a multipart body with the default
 * limits, staging every file with NO whitelist (so a file is recorded with its `mime` + `validated`
 * flag but never rejected on type). A non-multipart request passes through untouched (`next()`).
 *
 * - `limits` — the per-request DoS bound ({@link MultipartLimits}), enforced mid-stream.
 * - `allowed` — a MIME whitelist (anti-spoofing / anti-polyglot). When set, the SNIFFED type is
 *   AUTHORITATIVE: each file's REAL type (sniffed from its leading magic bytes) must be on the list AND
 *   agree with its declared `Content-Type` — so a declared type that doesn't match the magic bytes (a
 *   polyglot / type-confusion upload, e.g. `<script>` bytes declared `image/png`), or a real type not on
 *   the list, REJECTS the file (its temp file is cleaned + it is dropped from `files`). The decision
 *   rests on the bytes, never the client-controlled declared type. When OMITTED, every file is staged and
 *   its `mime` + `validated` recorded, but none is rejected on type. (An empty array allows NOTHING —
 *   every file is rejected.)
 * - `directory` — the directory to stage temp files in. Defaults to the OS temp dir (`os.tmpdir()`).
 */
export interface MultipartOptions {
	readonly limits?: MultipartLimits
	readonly allowed?: readonly string[]
	readonly directory?: string
}

/**
 * The parsed result of a `multipart/form-data` request body — what
 * {@link import('./middlewares.js').createMultipart} resolves the context's `body()` to.
 *
 * @remarks
 * The handler reads it via `await context.body()`, narrowed with a guard (`isRecord` /
 * {@link import('./helpers.js').isMultipartBody}, never `as` — §14). `files` holds the staged
 * {@link UploadedFileInterface}s (the app then `move`s / deletes them); `fields` holds the text field
 * values keyed by their form field name (a repeated field keeps the LAST value).
 */
export interface MultipartBody {
	readonly files: readonly UploadedFileInterface[]
	readonly fields: Readonly<Record<string, string>>
}

/**
 * One parsed part HEADER block of a multipart body — the decoded `Content-Disposition` +
 * `Content-Type` of a single part, the value {@link import('./helpers.js').parsePartHeaders}
 * returns from a part's raw header bytes.
 *
 * @remarks
 * - `name` — the form field name from `Content-Disposition` (`name="avatar"`); `undefined` when the
 *   disposition omits it (a malformed part).
 * - `filename` — the client filename from `Content-Disposition` (`filename="cat.png"`) when present;
 *   its PRESENCE distinguishes a FILE part (has a filename) from a FIELD part (none). Untrusted
 *   metadata — never used to build a path.
 * - `type` — the part's declared `Content-Type`, or `undefined` when the part omits one.
 */
export interface MultipartPart {
	readonly name: string | undefined
	readonly filename: string | undefined
	readonly type: string | undefined
}

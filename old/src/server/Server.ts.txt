import type { IncomingMessage, Server as NodeHTTPServer, ServerResponse } from 'node:http'
import type { Duplex } from 'node:stream'
import type { AbortInterface, EmitterInterface, TimeoutInterface } from '@src/core'
import type {
	Middleware,
	RouteGroupInterface,
	RouteHandler,
	RouteInput,
	ServerEventMap,
	ServerInterface,
	ServerOptions,
	ServerStatus,
	UpgradeHandler,
} from './types.js'
import { createServer as createHTTPServer } from 'node:http'
import { createAbort, createTimeout, Emitter } from '@src/core'
import { DEFAULT_DRAIN_MS } from './constants.js'
import { isAddressInfo } from './helpers.js'
import { MiddlewareManager } from './MiddlewareManager.js'
import { RequestManager } from './RequestManager.js'
import { RouteHandlerContext } from './routes/RouteHandlerContext.js'
import { RouteManager } from './routes/RouteManager.js'

/**
 * The HTTP server facade — an observable lifecycle over `node:http` composing a
 * route manager, a Koa-onion middleware chain, the core L1 `Abort` (stop signal)
 * and `Timeout` (drain grace).
 *
 * @remarks
 * - **Lifecycle (AGENTS §10).** `start()` builds the underlying `node:http` server,
 *   listens on the configured {@link ServerOptions.host} / {@link ServerOptions.port}
 *   (an undefined / `0` port binds an EPHEMERAL port — the default — resolved from the
 *   bound address via guards, an undefined host binds node's default interface; no
 *   PortManager), and transitions `idle → starting → listening`, resolving the bound
 *   port. `stop()` transitions to `stopping`: it
 *   stops accepting new requests, arms a drain {@link TimeoutInterface}, fires the
 *   stop {@link AbortInterface} so in-flight handlers observe cancellation, hands
 *   in-flight requests up to the `drain` deadline to finish (event-driven via the
 *   {@link RequestManager}, no busy-loop), then closes the socket → `stopped`.
 *   `destroy()` is the final teardown (abort + close + emitter destroy), idempotent.
 * - **Per request.** A fresh {@link RouteHandlerContext} is built; the composed
 *   middleware onion runs, ending in the matched route's handler (or a 404 / 405 /
 *   OPTIONS answer). A `HEAD` runs the matching `GET` handler and the body is
 *   discarded by `node:http`. Every unhandled throw is caught so a handler error
 *   becomes a 500 — the {@link createErrorBoundary} middleware is the real handler,
 *   but the server itself never crashes on an escaping throw.
 * - **Upgrade fan-out.** A raw `node:http` protocol-upgrade is fanned out to the
 *   handlers registered via `upgrade(...)` in order; the FIRST {@link UpgradeHandler}
 *   to return `true` CLAIMS (owns) the socket, an unclaimed upgrade has its socket
 *   DESTROYED (no dangling-connection leak), and an `upgrade` event reports the outcome.
 *   A handler that THROWS is caught and treated as declined and surfaced on the `error`
 *   event (symmetric with the request path's escaping-throw handling) so a buggy handler
 *   can never crash the process or leak the socket. The spine stays protocol-agnostic — it knows nothing of WebSocket; a transport that
 *   does attaches through this seam.
 * - **Observable (§13).** Owns an {@link Emitter} over {@link ServerEventMap}
 *   exposed as `readonly emitter`; lifecycle moments (`start` / `stop` / `request` /
 *   `upgrade` / `error` / `drain`) are emitted directly; the emitter isolates a listener
 *   throw and routes it to its `error` handler (the `error` option), so a buggy observer can
 *   never crash the server or perturb the lifecycle (`error` here is a DOMAIN event — a
 *   server-level fault). It adds no new concurrency primitive — the wake-park drain is the
 *   {@link RequestManager}.
 */
export class Server implements ServerInterface {
	readonly id = crypto.randomUUID()
	readonly #drain: number
	readonly #routes = new RouteManager()
	readonly #middleware = new MiddlewareManager()
	readonly #upgradeHandlers: UpgradeHandler[] = []
	readonly #requests = new RequestManager()
	readonly #emitter: Emitter<ServerEventMap>
	#http: NodeHTTPServer | undefined
	#abort: AbortInterface = createAbort()
	#status: ServerStatus = 'idle'
	#port: number | undefined
	readonly #host: string | undefined
	readonly #configuredPort: number | undefined

	constructor(options?: ServerOptions) {
		this.#drain = Math.max(0, options?.drain ?? DEFAULT_DRAIN_MS)
		this.#host = options?.host
		this.#configuredPort = options?.port
		this.#emitter = new Emitter<ServerEventMap>({ on: options?.on, error: options?.error })
	}

	get status(): ServerStatus {
		return this.#status
	}

	get emitter(): EmitterInterface<ServerEventMap> {
		return this.#emitter
	}

	get port(): number | undefined {
		return this.#port
	}

	use(middleware: Middleware): void {
		this.#middleware.use(middleware)
	}

	upgrade(handler: UpgradeHandler): void {
		this.#upgradeHandlers.push(handler)
	}

	route(inputs: readonly RouteInput[]): void
	route(input: RouteInput): void
	route(input: RouteInput | readonly RouteInput[]): void {
		this.#routes.add(Array.isArray(input) ? input : [input])
	}

	group(prefix: string): RouteGroupInterface {
		return this.#routes.group(prefix)
	}

	start(): Promise<number> {
		if (this.#status !== 'idle' && this.#status !== 'stopped') {
			return Promise.reject(new Error(`server cannot start from '${this.#status}'`))
		}
		this.#status = 'starting'
		// A fresh stop signal per run, so a restarted server is not born aborted.
		this.#abort = createAbort()
		const server = createHTTPServer((request, response) => this.#handle(request, response))
		// Fan a raw `node:http` protocol-upgrade out to the registered handlers. The
		// listener is bound to THIS run's server instance, which `stop` / `destroy`
		// discard (set `#http = undefined`), so it is collected with the closed server —
		// no manual removal needed across a stop / restart (the same per-run lifecycle as
		// the request handler above).
		server.on('upgrade', (request, socket, head) => this.#onUpgrade(request, socket, head))
		this.#http = server
		return new Promise<number>((resolve, reject) => {
			const onError = (error: Error): void => {
				server.off('listening', onListening)
				this.#http = undefined
				this.#status = 'idle'
				reject(error)
			}
			const onListening = (): void => {
				server.off('error', onError)
				const port = this.#resolvePort(server)
				this.#port = port
				this.#status = 'listening'
				this.#emitter.emit('start', port)
				resolve(port)
			}
			server.once('error', onError)
			server.once('listening', onListening)
			// Bind the configured host + port; an undefined / `0` port → an ephemeral
			// OS-assigned port (the default), an undefined host → node's default interface.
			// The actually-bound port is read from `address()` in `onListening` (so an
			// ephemeral `0` still resolves to the real port). A port already in use surfaces
			// as `EADDRINUSE` on the `error` listener above — see `discoverPort` to pre-pick a
			// guaranteed-free port.
			server.listen(this.#configuredPort ?? 0, this.#host)
		})
	}

	async stop(): Promise<void> {
		if (this.#status !== 'listening') return
		this.#status = 'stopping'
		this.#emitter.emit('stop')
		const server = this.#http
		// Fire the stop signal so in-flight handlers observing it can bail. It is a
		// pure signal — NOT the drain deadline's parent (a parent abort would CLEAR
		// the Timeout so it never fires). The drain deadline is an independent clock:
		// the request manager parks on its signal (event-driven, no busy-loop) and
		// resolves when the last request finishes OR the deadline expires.
		this.#abort.abort()
		const deadline: TimeoutInterface = createTimeout({ ms: this.#drain })
		deadline.start()
		await this.#requests.drain(deadline.signal)
		deadline.clear()
		const pending = this.#requests.count
		this.#emitter.emit('drain', pending)
		// Force the sockets closed only if the deadline fired with work still in
		// flight; a clean drain (`pending === 0`) closes gracefully.
		if (server !== undefined) await this.#close(server, pending > 0)
		this.#http = undefined
		this.#port = undefined
		this.#status = 'stopped'
	}

	async destroy(): Promise<void> {
		if (this.#status === 'stopped' && this.#http === undefined) {
			this.#emitter.destroy()
			return
		}
		if (!this.#abort.aborted) this.#abort.abort()
		const server = this.#http
		// Final teardown — destroy any lingering sockets so close never hangs.
		if (server !== undefined) await this.#close(server, true)
		this.#http = undefined
		this.#port = undefined
		this.#status = 'stopped'
		this.#emitter.destroy()
	}

	// Build the per-request context, track the request for draining, and run the
	// middleware onion ending in the route dispatch — guarding every escaping throw
	// so the process never crashes on an unhandled handler error.
	#handle(request: IncomingMessage, response: ServerResponse): void {
		const finish = this.#requests.start()
		response.once('finish', finish)
		response.once('close', finish)
		const context = new RouteHandlerContext(request, response)
		this.#emitter.emit('request', context.method, context.url.pathname)
		const runner = this.#middleware.compose(this.#dispatch())
		Promise.resolve(runner(context)).catch((error: unknown) => {
			this.#emitter.emit('error', error)
			if (!response.headersSent) context.error(500, 'Internal Server Error')
			else if (!response.writableEnded) response.end()
		})
	}

	// Fan a raw protocol-upgrade out to the registered handlers in registration order:
	// the FIRST to return `true` CLAIMS the socket (it now owns the connection — stop).
	// A handler that THROWS is caught and treated as DECLINED — the throw is surfaced on
	// the `error` event (the SAME channel an escaping route-handler throw uses in `#handle`)
	// and the loop continues so a later handler can still claim, mirroring the request
	// path's throw-isolation: a buggy handler can NEVER crash the process (the `'upgrade'`
	// listener runs raw on `node:http`, so an uncaught throw here would be an
	// uncaughtException — crashing the server AND leaking the socket). If NONE claim it,
	// DESTROY the socket so an unhandled upgrade never leaks a dangling connection. The
	// `upgrade` event fires AFTER the fan-out settles (carrying whether a handler claimed
	// it) — observation-only, and the emitter isolates a listener throw, so a buggy observer can
	// neither claim the socket nor perturb this decision. The spine speaks NO upgrade
	// protocol itself: `request` / `socket` / `head` are node's own `IncomingMessage` /
	// `Duplex` / `Buffer` (no assertion at the boundary — §14), handed verbatim to the
	// handler that does.
	#onUpgrade(request: IncomingMessage, socket: Duplex, head: Buffer): void {
		let handled = false
		for (const handler of this.#upgradeHandlers) {
			try {
				if (handler(request, socket, head)) {
					handled = true
					break
				}
			} catch (error) {
				// A throwing handler is treated as declined: surface it on `error` (as the
				// request path does) and let a later handler still claim — never crash.
				this.#emitter.emit('error', error)
			}
		}
		if (!handled) socket.destroy()
		this.#emitter.emit('upgrade', request, handled)
	}

	// The terminal handler the onion wraps: resolve the route and run its handler,
	// or answer OPTIONS / 405 / 404 directly.
	#dispatch(): RouteHandler {
		return async (context) => {
			if (context.method === 'OPTIONS') {
				const allow = this.#routes.allow(context.url.pathname)
				if (allow.length > 0) {
					context.response.setHeader('Allow', [...allow, 'OPTIONS'].join(', '))
					context.empty(204)
					return
				}
			}
			const match = this.#routes.match(context.method, context.url.pathname)
			if (match.status === 'matched') {
				// Thread the matched params onto the concrete context — narrowed via
				// `instanceof` (the spine always builds a `RouteHandlerContext`), no `as`.
				if (context instanceof RouteHandlerContext) context.assign(match.params)
				await match.route.handler(context)
				return
			}
			if (match.status === 'unmethoded') {
				context.response.setHeader('Allow', match.allow.join(', '))
				context.error(405)
				return
			}
			context.error(404)
		}
	}

	// Resolve the bound port from the server's address, narrowed via a guard (an
	// `AddressInfo` object carries a numeric `port`; a string / null address has no
	// port). No assertion — an unresolvable address yields `0`.
	#resolvePort(server: NodeHTTPServer): number {
		const address = server.address()
		if (isAddressInfo(address)) {
			return address.port
		}
		return 0
	}

	// Close the underlying server, resolving once it stops accepting connections.
	// `server.close()` waits for open connections to end — and a keep-alive client
	// (undici `fetch`) leaves its socket IDLE after a response, which would hang the
	// close. So always drop IDLE sockets (an in-flight request is untouched); when
	// `force` is set (the drain deadline fired with requests still in flight, or
	// `destroy`) DESTROY every open socket so the callback fires promptly rather than
	// hanging on a stuck handler.
	#close(server: NodeHTTPServer, force: boolean): Promise<void> {
		return new Promise<void>((resolve) => {
			server.close(() => resolve())
			if (force) server.closeAllConnections()
			else server.closeIdleConnections()
		})
	}
}

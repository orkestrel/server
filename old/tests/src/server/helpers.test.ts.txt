describe('discoverPort â€” free TCP port probe', () => {
	const { track } = createTeardown((handle: StartedServerInterface) => handle.stop())

	it('returns a free, usable ephemeral port when none is preferred', async () => {
		const port = await discoverPort()
		expect(port).toBeGreaterThan(0)
		// The probe released it â€” a real server can bind it (no lingering listener).
		const server = createServer({ port })
		const handle = track(await startServer(server))
		expect(handle.port).toBe(port)
	})

	it('returns the preferred port when it is free', async () => {
		// First discover a free port, then ASK for that exact one â€” it must be returned
		// unchanged (it was free at the first probe and still is).
		const preferred = await discoverPort()
		expect(await discoverPort(preferred)).toBe(preferred)
	})

	it('falls back to an ephemeral port when the preferred one is taken', async () => {
		// Occupy a real port with a throwaway net server, then ask discoverPort for it: the
		// EADDRINUSE must trigger the fallback to a DIFFERENT free ephemeral port (never the
		// taken one, never a reject). The blocker binds the SAME scope discoverPort's probe
		// uses (all interfaces â€” no host), so the bind genuinely collides; a loopback-only
		// blocker would NOT conflict with the probe's wildcard bind.
		const taken = await discoverPort()
		const blocker = createNetServer()
		await new Promise<void>((done) => blocker.listen(taken, done))
		try {
			const fallback = await discoverPort(taken)
			expect(fallback).toBeGreaterThan(0)
			expect(fallback).not.toBe(taken)
		} finally {
			await new Promise<void>((done) => blocker.close(() => done()))
		}
	})
})

// signToken / verifyToken â€” the stateless HMAC-signed token primitives behind
// createTokenGuard. Pure + deterministic (no server): a round-trip returns the original
// value; tampering (signature OR payload) is rejected; a non-token / no-dot string is
// rejected; rotation accepts an older secret; an HMAC-bound expiry rejects a past token
// (and a realistic short-timer path); a value containing dots survives the lastIndexOf
// split; an empty secret list makes verify total-`undefined` and signToken throw.
describe('signToken / verifyToken â€” stateless signed tokens', () => {
	it('round-trips: a signed value verifies back to the original', () => {
		const token = signToken('client', { secret: 'secret-k' })
		expect(verifyToken(token, 'secret-k')).toBe('client')
	})

	it('rejects a tampered signature', () => {
		const token = signToken('client', { secret: 'secret-k' })
		// Flip the last char of the signature (after the dot) â€” the HMAC no longer matches.
		const last = token.endsWith('a') ? 'b' : 'a'
		const tampered = token.slice(0, -1) + last
		expect(verifyToken(tampered, 'secret-k')).toBeUndefined()
	})

	it('rejects a tampered payload (re-encoding a different value under the old signature)', () => {
		const token = signToken('client', { secret: 'secret-k' })
		const signature = token.slice(token.lastIndexOf('.') + 1)
		// Swap in a payload encoding a DIFFERENT value but keep the original signature â€”
		// the recomputed HMAC over the new payload won't match.
		const forgedPayload = Buffer.from(JSON.stringify({ value: 'admin' })).toString('base64url')
		expect(verifyToken(`${forgedPayload}.${signature}`, 'secret-k')).toBeUndefined()
	})

	it('rejects a non-token string with no dot', () => {
		expect(verifyToken('not-a-token', 'secret-k')).toBeUndefined()
		expect(verifyToken('', 'secret-k')).toBeUndefined()
	})

	it('rejects a wrong secret', () => {
		const token = signToken('client', { secret: 'secret-k' })
		expect(verifyToken(token, 'other-secret')).toBeUndefined()
	})

	it('rotation: a token signed with A verifies under [B, A] but not under [B] alone', () => {
		// signToken always signs with the FIRST secret (A here); verifyToken accepts ANY in
		// the list â€” so a rotation list with the old secret still present accepts it, and
		// dropping it (only the new B) rejects it.
		const token = signToken('client', { secret: 'secret-a' })
		expect(verifyToken(token, ['secret-b', 'secret-a'])).toBe('client')
		expect(verifyToken(token, ['secret-b'])).toBeUndefined()
	})

	it('signs with the current (first) secret of a rotation list', () => {
		// A rotation list signs with its head ('current'); verifying with that head alone works.
		const token = signToken('client', { secret: ['current', 'older'] })
		expect(verifyToken(token, 'current')).toBe('client')
		expect(verifyToken(token, 'older')).toBeUndefined()
	})

	it('expiry: an already-past ttl rejects, a generous ttl verifies', () => {
		const expired = signToken('client', { secret: 'secret-k', ttl: -1000 })
		expect(verifyToken(expired, 'secret-k')).toBeUndefined()
		const live = signToken('client', { secret: 'secret-k', ttl: 10_000 })
		expect(verifyToken(live, 'secret-k')).toBe('client')
	})

	it('expiry: a short real timer elapses and the token stops verifying', async () => {
		const token = signToken('client', { secret: 'secret-k', ttl: 10 })
		expect(verifyToken(token, 'secret-k')).toBe('client')
		await waitForDelay(20)
		expect(verifyToken(token, 'secret-k')).toBeUndefined()
	})

	it('a value containing dots survives the round-trip (split on the LAST dot)', () => {
		const token = signToken('a.b.c', { secret: 'secret-k' })
		expect(verifyToken(token, 'secret-k')).toBe('a.b.c')
	})

	it('an empty secret list: verify is total-undefined, signToken throws', () => {
		expect(verifyToken('anything.signature', [])).toBeUndefined()
		expect(() => signToken('client', { secret: [] })).toThrow(HTTPError)
	})

	it('a blank / whitespace-only secret is fail-closed on sign (throws)', () => {
		// No usable (non-blank) secret left after normalization â†’ the empty-list guard
		// trips, so a publicly-computable empty-key HMAC can never sign.
		expect(() => signToken('client', { secret: '' })).toThrow(HTTPError)
		expect(() => signToken('client', { secret: '   ' })).toThrow(HTTPError)
		expect(() => signToken('client', { secret: ['', '  '] })).toThrow(HTTPError)
	})

	it('a blank secret is fail-closed on verify (rejects a real-secret token)', () => {
		// A token signed with a REAL secret must NOT verify under a blank secret â€” the
		// blank is dropped, leaving nothing usable, so verify is total-undefined.
		const token = signToken('client', { secret: 'secret-k' })
		expect(verifyToken(token, '')).toBeUndefined()
	})

	it('a blank entry in a rotation list is ignored â€” a real secret still verifies', () => {
		// A blank alongside a real secret must NOT break a valid token: the blank is
		// dropped and the real secret still matches.
		const token = signToken('client', { secret: 'secret-k' })
		expect(verifyToken(token, ['', 'secret-k'])).toBe('client')
	})
})

// normalizeSecret / safeCompare â€” the token internals behind signToken / verifyToken,
// now exported. normalizeSecret resolves a TokenSecret to the usable (non-blank) list;
// safeCompare is the length-guarded constant-time compare (it must NOT throw on a

describe('normalizeSecret â€” TokenSecret â†’ usable secret list', () => {
	it('wraps a single string in a one-element list', () => {
		expect([...normalizeSecret('secret-k')]).toEqual(['secret-k'])
	})

	it('copies a rotation list verbatim', () => {
		const list = ['current', 'older']
		const out = normalizeSecret(list)
		expect([...out]).toEqual(['current', 'older'])
	})

	it('drops blank / whitespace-only entries (keeping the real ones, verbatim)', () => {
		expect([...normalizeSecret(['', '  ', 'secret-k', '\t'])]).toEqual(['secret-k'])
	})

	it('an all-blank secret normalizes to the empty list (fail-closed)', () => {
		expect([...normalizeSecret('')]).toEqual([])
		expect([...normalizeSecret('   ')]).toEqual([])
		expect([...normalizeSecret(['', '  '])]).toEqual([])
	})
})

describe('safeCompare â€” length-guarded constant-time compare', () => {
	it('equal strings compare true', () => {
		expect(safeCompare('abc', 'abc')).toBe(true)
	})

	it('unequal strings of the SAME length compare false', () => {
		expect(safeCompare('abc', 'abd')).toBe(false)
	})

	it('unequal-LENGTH strings compare false WITHOUT throwing (the length guard)', () => {
		// `timingSafeEqual` throws on a length mismatch; the guard returns false instead.
		expect(safeCompare('abc', 'abcd')).toBe(false)
		expect(safeCompare('', 'x')).toBe(false)
	})
})

//
// src/server/http/helpers.ts â€” the cookie helpers behind the signed-cookie session
// transport (`createCookieSession`) and `createCSRFGuard`. `parseCookies` /
// `decodeCookieValue` / `serializeCookie` are pure (plain string in/out); the SIGNED
// pair (`readSignedCookie`, paired with `signToken` for the wire value) is exercised
// against a request STUB carrying a `Cookie` header (no `as`, Â§14) â€” proving the HMAC
// signâ†”verify round-trip AND tamper rejection without a server (the over-the-wire
// `Set-Cookie` writes are proven through a real cookie session in middlewares.test.ts).

describe('parseCookies â€” raw Cookie header â†’ name/value lookup', () => {
	it('parses a multi-cookie header, trimming whitespace around each pair', () => {
		expect(parseCookies('session=abc; theme=dark; lang=en')).toEqual({
			session: 'abc',
			theme: 'dark',
			lang: 'en',
		})
	})

	it('URL-decodes the value (the inverse of serializeCookie encoding)', () => {
		// A value with reserved characters round-trips: `a b/c` â†’ `a%20b%2Fc` on the wire â†’ decoded back.
		expect(parseCookies('x=a%20b%2Fc')).toEqual({ x: 'a b/c' })
	})

	it('keeps a quoted value verbatim (quotes are part of the value, not stripped)', () => {
		// RFC 6265 quoted cookie-values: the surrounding quotes are passed through (decodeURIComponent
		// leaves them), so a reader sees the literal quoted text â€” not unwrapped.
		expect(parseCookies('q="spaced value"')).toEqual({ q: '"spaced value"' })
	})

	it('skips a pair with no = and an empty name; an absent / empty header is {}', () => {
		expect(parseCookies('flag; =orphan; ok=1')).toEqual({ ok: '1' })
		expect(parseCookies('')).toEqual({})
		expect(parseCookies(undefined)).toEqual({})
	})

	it('a later duplicate name wins', () => {
		expect(parseCookies('a=1; a=2')).toEqual({ a: '2' })
	})

	// SECURITY (the __Host- / __Secure- prefix-integrity fix): a cookie NAME is NOT trimmed into
	// compliance â€” a name carrying leading / trailing (or any) whitespace, or any char outside the
	// RFC 6265 cookie-token set, is REJECTED rather than reconciled, so a `'  __Host-x'` cannot
	// silently become the prefix-protected `__Host-x`.
	it('does NOT trim a leading-whitespace name into a __Host- / __Secure- prefix', () => {
		// The attacker plants `  __Host-session=evil` hoping the trim makes it the protected name. The
		// key is NOT `__Host-session` (the whitespace name is dropped); a browser would reject this
		// `__Host-`-prefixed cookie outright, and our parser must not undo that by trimming.
		const hostAttack = parseCookies('  __Host-session=evil')
		expect(hostAttack['__Host-session']).toBeUndefined()
		expect(Object.keys(hostAttack)).not.toContain('__Host-session')

		const secureAttack = parseCookies(' __Host-x=evil')
		expect(secureAttack['__Host-x']).toBeUndefined()
		expect(Object.keys(secureAttack)).not.toContain('__Host-x')
	})

	it('rejects a name with interior or trailing whitespace, or a non-token char', () => {
		// Trailing whitespace, interior whitespace, and a separator char are all outside the cookie
		// token set â€” each pair is skipped (not reconciled), keeping a clean cookie alongside it intact.
		expect(parseCookies('bad name=1; ok=2')).toEqual({ ok: '2' }) // interior space â†’ name rejected
		expect(parseCookies('trailing\t=1; ok=2')).toEqual({ ok: '2' }) // trailing tab â†’ rejected
		expect(parseCookies('a(b=1; ok=2')).toEqual({ ok: '2' }) // `(` is a separator â†’ rejected
	})

	it('rejects a name padded with a Unicode (non-ASCII) space â€” the prefix cannot be smuggled', () => {
		// U+00A0 (no-break space) is NOT in the cookie token set and IS matched by the name guard's
		// rejection â€” so a Unicode-space-padded `__Host-` name is dropped, not trimmed into the prefix.
		const nbsp = 'Â '
		const attack = parseCookies(`${nbsp}__Host-id=evil`)
		expect(attack['__Host-id']).toBeUndefined()
	})

	it('still accepts a clean prefix-named cookie (the guard does not over-reject)', () => {
		// A correctly-formed `__Host-` cookie (no surrounding whitespace) parses normally â€” the fix
		// rejects only WHITESPACE-padded / malformed names, never a legitimate prefixed one.
		expect(parseCookies('__Host-session=good; __Secure-id=ok')).toEqual({
			'__Host-session': 'good',
			'__Secure-id': 'ok',
		})
	})
})

describe('serializeCookie â€” name/value + attributes â†’ Set-Cookie value', () => {
	it('defaults to Path=/, HttpOnly, SameSite=Lax (no Secure, no Max-Age)', () => {
		expect(serializeCookie('session', 'abc')).toBe('session=abc; Path=/; HttpOnly; SameSite=Lax')
	})

	it('URL-encodes the value', () => {
		expect(serializeCookie('x', 'a b/c')).toBe('x=a%20b%2Fc; Path=/; HttpOnly; SameSite=Lax')
	})

	it('emits the present attributes in canonical order (Domain, Path, Max-Age, HttpOnly, Secure, SameSite)', () => {
		expect(
			serializeCookie('s', 'v', {
				domain: 'example.test',
				path: '/app',
				maxAge: 3600,
				secure: true,
				sameSite: 'Strict',
			}),
		).toBe('s=v; Domain=example.test; Path=/app; Max-Age=3600; HttpOnly; Secure; SameSite=Strict')
	})

	it('omits HttpOnly only with an explicit httpOnly:false (the CSRF-cookie shape)', () => {
		expect(serializeCookie('csrf', 't', { httpOnly: false, sameSite: 'Strict' })).toBe(
			'csrf=t; Path=/; SameSite=Strict',
		)
	})

	it('emits Max-Age=0 (the immediate-expiry clear form), distinct from an omitted Max-Age', () => {
		expect(serializeCookie('s', '', { maxAge: 0 })).toBe(
			's=; Path=/; Max-Age=0; HttpOnly; SameSite=Lax',
		)
	})

	// SECURITY (attribute-injection fix): the `Domain` / `Path` values are validated, not interpolated
	// raw â€” a value carrying `;` / `,` / whitespace / a control char could inject another `Set-Cookie`
	// attribute or split the header, so it THROWS a clear error (a programmer misconfiguration, Â§12).
	it('rejects a Domain carrying an injected attribute (`; Secure`)', () => {
		expect(() => serializeCookie('s', 'v', { domain: 'evil.test; Secure' })).toThrow(/Domain/)
	})

	it('rejects a Path carrying a header-splitting / control sequence', () => {
		expect(() => serializeCookie('s', 'v', { path: '/a\r\nSet-Cookie: x=y' })).toThrow(/Path/)
		expect(() => serializeCookie('s', 'v', { path: '/a b' })).toThrow(/Path/) // a space is illegal
	})

	it('accepts a clean Domain / Path (the guard does not over-reject normal values)', () => {
		expect(serializeCookie('s', 'v', { domain: 'sub.example.test', path: '/app/v2' })).toBe(
			's=v; Domain=sub.example.test; Path=/app/v2; HttpOnly; SameSite=Lax',
		)
	})

	// SECURITY (SameSite=None requires Secure): a `None` cookie is rejected by browsers unless it is
	// `Secure`, so serializeCookie FORCES `Secure` on a `None` cookie even when `secure` is unset / false.
	it('forces Secure when sameSite is None (even with secure unset or false)', () => {
		expect(serializeCookie('s', 'v', { sameSite: 'None' })).toBe(
			's=v; Path=/; HttpOnly; Secure; SameSite=None',
		)
		// An explicit `secure: false` cannot defeat the spec requirement â€” `None` is still `Secure`.
		expect(serializeCookie('s', 'v', { sameSite: 'None', secure: false })).toContain('Secure')
	})

describe('readSignedCookie + signToken â€” HMAC cookie sign â†” verify + tamper rejection', () => {

	// cookie (the exact wire serializeCookie/writeSignedCookie produce) â€” no server needed (Â§14/Â§16).
	function requestWithCookie(cookie: string): ReturnType<typeof createRequestStub> {
		return createRequestStub({ headers: { cookie } })
	}

	it('round-trips: a signed cookie value verifies back to the embedded value', () => {

		const wire = signToken('session-7', { secret: SECRET })

// The GENERIC SSE seam (openSSEStream + bindStreamingAbort) over a REAL node:http
// server (no mocks, the node src:server project). Each test stands a server on an
// ephemeral port, drives it with real fetch, decodes the response body with the core
// SSEParser (collectSSE / readSSEStream), and stops + destroys it in afterEach so no
// socket leaks. The decode-with-SSEParser is the proof the wire round-trips: every
// assertion is on the events the parser dispatched from the bytes the seam serialized â€”
// the encode â†” decode inverse. Covers: SSE headers, single + multiline data,
// event/id/retry fields, comment keep-alives, write-after-end no-op, ordering, the
// disconnectâ†’once-only-abort binding + leak-free cleanup, and the graceful-drain
// interaction (an open stream + server.stop() resolves, never hangs).

const { track } = createTeardown((handle: StartedServerInterface) => handle.stop())

// A server whose single GET /sse route runs `drive` against a fresh SSE stream â€” the
// shared scaffold for the deterministic seam tests. `drive` writes events then ends.

	const server = createServer()
	server.route({
		method: 'GET',
		path: '/sse',
		handler: (context) => {
			drive(openSSEStream(context))
		},
	})
	return server
}

describe('bindStreamingAbort â€” disconnect â†’ once-only abort', () => {
	it('a client aborting the fetch fires onAbort exactly once, server-side', async () => {
		const aborts = createRecorder<readonly [unknown]>()
		const server = createServer()
		server.route({
			method: 'GET',
			path: '/sse',
			handler: async (context) => {
				const sse = openSSEStream(context)
				bindStreamingAbort(context, (reason) => aborts.handler(reason))
				// Stream slowly and indefinitely until the client disconnects â€” never end
				// server-side, so the ONLY thing that stops it is the client abort.
				for (let n = 0; n < 200 && !sse.closed; n += 1) {
					sse.write({ event: 'tick', data: String(n) })
					await waitForDelay(15)
				}
			},
		})
		const handle = track(await startServer(server))
		const controller = new AbortController()
		const response = await fetch(`${handle.base}/sse`, { signal: controller.signal })

		// Consume a couple of real events, then abort the client fetch mid-stream.
		let seen = 0
		try {
			for await (const event of readSSEStream(response)) {
				void event
				seen += 1
				if (seen >= 2) {
					controller.abort()
					break
				}
			}
		} catch {
			// The reader rejecting on abort is expected once we cancel â€” not a failure.
		}
		expect(seen).toBeGreaterThanOrEqual(2)

		// Give the server a moment to observe the socket close and fire onAbort.
		await waitForDelay(150)
		// Fired â€” and exactly once (the request close + response close are coalesced).
		expect(aborts.count).toBe(1)
	})

	it('the cleanup fn removes the listeners â€” a normal end never fires onAbort', async () => {
		const aborts = createRecorder<readonly [unknown]>()
		const handle = track(
			await startServer(
				(() => {
					const server = createServer()
					server.route({
						method: 'GET',
						path: '/sse',
						handler: (context) => {
							const sse = openSSEStream(context)
							const cleanup = bindStreamingAbort(context, (reason) => aborts.handler(reason))
							sse.write({ data: 'done' })
							sse.end()
							// The stream ended normally â€” detach the disconnect listeners so the
							// response's own 'close' does NOT later fire onAbort (no leak).
							cleanup()
						},
					})
					return server
				})(),
			),
		)
		const events = await collectSSE(await fetch(`${handle.base}/sse`))
		expect(events).toEqual([{ data: 'done' }])
		// Let any (removed) close listeners have fired â€” they must not have.
		await waitForDelay(100)
		expect(aborts.count).toBe(0)
	})
})

describe('graceful drain â€” an open SSE stream + server.stop() resolves', () => {
	it('does not hang: stop fires, the bound abort ends the stream, drain completes', async () => {
		const aborts = createRecorder<readonly [unknown]>()
		const server = createServer({ drain: 2_000 })
		server.route({
			method: 'GET',
			path: '/sse',
			handler: async (context) => {
				const sse = openSSEStream(context)
				const cleanup = bindStreamingAbort(context, (reason) => aborts.handler(reason))
				// The CLEAN drain path: observe the server's stop event and end the stream,
				// so stop â†’ end() â†’ response finishes â†’ request drains, with no force-close
				// wait. (A consumer wires this; the seam stays agent-agnostic.) Disconnect
				// also ends it via bindStreamingAbort.
				const onStop = (): void => {
					aborts.handler('stop')
					sse.end()
				}
				server.emitter.on('stop', onStop)
				// Stream indefinitely until the stream is closed (by the stop handler above
				// or a client disconnect).
				for (let n = 0; n < 500 && !sse.closed; n += 1) {
					sse.write({ event: 'tick', data: String(n) })
					await waitForDelay(15)
				}
				server.emitter.off('stop', onStop)
				cleanup()
			},
		})
		const handle = track(await startServer(server))
		// Start consuming so the request is genuinely in flight on the server.
		const response = await fetch(`${handle.base}/sse`)
		const reader = readSSEStream(response)
		await reader.next() // pull at least one event so the handler is mid-stream

		// Stop with an open stream â€” this must RESOLVE (the stop event ends the stream,
		// the request drains), well within the drain deadline, never hanging.
		const before = Date.now()
		await server.stop()
		const elapsed = Date.now() - before
		expect(server.status).toBe('stopped')
		// Resolved promptly via the clean stopâ†’end path â€” far under the 2s drain deadline
		// (which would only matter if the stream had NOT self-terminated).
		expect(elapsed).toBeLessThan(1_500)
		// The stop handler observed the stop and ended the stream.
		expect(aborts.calls.some((call) => call[0] === 'stop')).toBe(true)

		// Drain the (now-ending) client stream so no reader is left dangling.
		try {
			for await (const event of reader) void event
		} catch {
			// The stream ending under us as the socket closes is expected.
		}
		// The handle's afterEach stop() is a no-op now (already stopped) â€” safe.
	})
})

// The static-file helpers â€” the PURE pieces createStatic composes (AGENTS Â§5): the
// SECURITY-CRITICAL traversal guard, the dotfile predicate, the MIME lookup, the Range
// parser, and the weak-ETag computer. Exercised directly on plain values (no FS, no
// server) â€” the over-the-wire serving is proven in middlewares.test.ts.

describe('resolveStaticPath â€” the traversal guard', () => {
	const root = resolve('/srv/app/public')

	it('resolves a plain in-root path to its absolute path under root', () => {
		expect(resolveStaticPath(root, '/app.js')).toBe(resolve(root, 'app.js'))
		expect(resolveStaticPath(root, '/assets/main.css')).toBe(resolve(root, 'assets/main.css'))
	})

	it('maps the bare root path to the root directory', () => {
		expect(resolveStaticPath(root, '/')).toBe(root)
	})

	it('decodes percent-escapes before resolving', () => {
		expect(resolveStaticPath(root, '/a%20b.txt')).toBe(resolve(root, 'a b.txt'))
	})

	it('BLOCKS a literal ../ escape (returns undefined â€” never resolves outside root)', () => {
		expect(resolveStaticPath(root, '/../secret.txt')).toBeUndefined()
		expect(resolveStaticPath(root, '/../../etc/passwd')).toBeUndefined()
		expect(resolveStaticPath(root, '/assets/../../secret')).toBeUndefined()
	})

	it('BLOCKS an ENCODED ..%2f escape (the decode happens before normalization)', () => {
		expect(resolveStaticPath(root, '/%2e%2e/secret')).toBeUndefined()
		expect(resolveStaticPath(root, '/..%2fsecret')).toBeUndefined()
		expect(resolveStaticPath(root, '/%2e%2e%2f%2e%2e%2fetc%2fpasswd')).toBeUndefined()
	})

	it('BLOCKS a NUL-byte poison path', () => {
		expect(resolveStaticPath(root, '/app.js%00.png')).toBeUndefined()
	})

	it('rejects a malformed percent-escape rather than throwing (TOTAL)', () => {
		expect(resolveStaticPath(root, '/%')).toBeUndefined()
		expect(resolveStaticPath(root, '/%zz')).toBeUndefined()
	})

	it('does not treat a sibling dir that shares the root prefix as in-root', () => {
		// `/srv/app/public-secret` starts with the root STRING but is NOT under `root + sep`.
		const sibling = resolve('/srv/app/public-secret/data')
		expect(resolveStaticPath(root, '/data')).not.toBe(sibling)
		expect(resolveStaticPath(root, '/data')).toBe(resolve(root, 'data'))
	})

	describe('with a prefix', () => {
		it('strips the prefix and resolves the remainder under root', () => {
			expect(resolveStaticPath(root, '/assets/app.js', '/assets')).toBe(resolve(root, 'app.js'))
			expect(resolveStaticPath(root, '/assets', '/assets')).toBe(root)
		})

		it('falls through (undefined) for a path not under the prefix', () => {
			expect(resolveStaticPath(root, '/other/app.js', '/assets')).toBeUndefined()
			// A path that only shares the prefix as a string but is not a path segment boundary.
			expect(resolveStaticPath(root, '/assetsX/app.js', '/assets')).toBeUndefined()
		})

		it('still blocks a traversal that climbs out after the prefix', () => {
			expect(resolveStaticPath(root, '/assets/../../secret', '/assets')).toBeUndefined()
		})
	})
})

describe('parseRange â€” the Range header parser (TOTAL)', () => {
	it('returns undefined for an absent / non-bytes / multi-range header (serve the whole file)', () => {
		expect(parseRange(undefined, 100)).toBeUndefined()
		expect(parseRange('items=0-10', 100)).toBeUndefined()
		expect(parseRange('bytes=0-10, 20-30', 100)).toBeUndefined()
	})

	it('parses a closed range and clamps the end into the resource', () => {
		expect(parseRange('bytes=0-9', 100)).toEqual({ satisfiable: true, start: 0, end: 9 })
		expect(parseRange('bytes=10-19', 100)).toEqual({ satisfiable: true, start: 10, end: 19 })
		// End past the last byte clamps to size - 1.
		expect(parseRange('bytes=90-200', 100)).toEqual({ satisfiable: true, start: 90, end: 99 })
	})

	it('parses an open range (start-) to the end of the resource', () => {
		expect(parseRange('bytes=50-', 100)).toEqual({ satisfiable: true, start: 50, end: 99 })
	})

	it('parses a suffix range (-N) as the last N bytes', () => {
		expect(parseRange('bytes=-20', 100)).toEqual({ satisfiable: true, start: 80, end: 99 })
		// A suffix larger than the resource clamps the start to 0.
		expect(parseRange('bytes=-200', 100)).toEqual({ satisfiable: true, start: 0, end: 99 })
	})

	it('reports an unsatisfiable range (a start at or past the size) â†’ satisfiable:false', () => {
		expect(parseRange('bytes=100-110', 100)).toEqual({ satisfiable: false })
		expect(parseRange('bytes=200-', 100)).toEqual({ satisfiable: false })
		// Any concrete range over a zero-length resource is unsatisfiable.
		expect(parseRange('bytes=0-0', 0)).toEqual({ satisfiable: false })
		expect(parseRange('bytes=-5', 0)).toEqual({ satisfiable: false })
	})

	it('returns undefined for a malformed range rather than throwing', () => {
		expect(parseRange('bytes=', 100)).toBeUndefined()
		expect(parseRange('bytes=abc-def', 100)).toBeUndefined()
		expect(parseRange('bytes=10-5', 100)).toBeUndefined() // end before start
		expect(parseRange('bytes=-0', 100)).toBeUndefined() // a zero suffix is meaningless
	})

	it('bytes=0-0 on a non-empty resource is the single FIRST byte {start:0, end:0}', () => {
		// The smallest closed range â€” the first byte alone. End clamps to min(0, size-1) = 0.
		expect(parseRange('bytes=0-0', 100)).toEqual({ satisfiable: true, start: 0, end: 0 })
	})

	it('bytes=5-2 (end < start) is malformed â†’ undefined', () => {
		// A closed range whose end precedes its start is malformed (distinct from an unsatisfiable
		// past-the-end range) â€” served as the whole resource, never a reversed window.
		expect(parseRange('bytes=5-2', 100)).toBeUndefined()
	})

	it('an open bytes=0- against a ZERO-length resource is unsatisfiable', () => {
		// `start (0) >= size (0)` â†’ unsatisfiable (a 416), not a {0, -1} window. A zero-length resource
		// has no byte to open-range from.
		expect(parseRange('bytes=0-', 0)).toEqual({ satisfiable: false })
	})
})

describe('computeBodyETag â€” the content validator (S-e)', () => {
	const body = Buffer.from('the dynamic response body')
	// The expected digest, computed with node:crypto directly (cross-check, not a re-implementation).
	const sha = createHash('sha256').update(body).digest('hex')

	it('wraps a weak validator as W/"<hash>" and a strong one as "<hash>"', () => {
		expect(computeBodyETag(body, 'sha256', true)).toBe(`W/"${sha}"`)
		expect(computeBodyETag(body, 'sha256', false)).toBe(`"${sha}"`)
	})

	it('is content-derived â€” identical bytes hash identically, different bytes differ', () => {
		const same = computeBodyETag(Buffer.from('payload'), 'sha256', true)
		expect(computeBodyETag(Buffer.from('payload'), 'sha256', true)).toBe(same)
		expect(computeBodyETag(Buffer.from('payloads'), 'sha256', true)).not.toBe(same)
	})

	it('honors the algorithm (a sha1 tag differs from a sha256 tag)', () => {
		const sha1 = createHash('sha1').update(body).digest('hex')
		expect(computeBodyETag(body, 'sha1', true)).toBe(`W/"${sha1}"`)
		expect(computeBodyETag(body, 'sha1', true)).not.toBe(computeBodyETag(body, 'sha256', true))
	})

	it('hashes an empty body to a constant (the empty-digest)', () => {
		const empty = createHash('sha256').update(Buffer.alloc(0)).digest('hex')
		expect(computeBodyETag(Buffer.alloc(0), 'sha256', true)).toBe(`W/"${empty}"`)
	})
})

// â”€â”€ The content-negotiation + compression helpers (S-c) â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
//
// The ONE shared q-value parser + the coding negotiation + the compressible predicate + the
// `node:zlib` round-trip â€” exercised directly on plain values (AGENTS Â§16). The compress/decompress
// helpers are cross-checked against `node:zlib`'s own `gzipSync` / `gunzipSync` / `brotliDecompressSync`
// so the wire is proven (not a self-referential round-trip alone).

describe('parseAcceptHeader â€” the shared q-value parser', () => {
	it('parses values + q-weights and sorts by q descending', () => {
		expect(parseAcceptHeader('text/html, application/json;q=0.9, text/plain;q=0.1')).toEqual([
			{ value: 'text/html', q: 1 },
			{ value: 'application/json', q: 0.9 },
			{ value: 'text/plain', q: 0.1 },
		])
	})

	it('lower-cases values, defaults a missing q to 1, and clamps to [0, 1]', () => {
		expect(parseAcceptHeader('GZIP, BR;q=2.0, deflate;q=-1')).toEqual([
			{ value: 'gzip', q: 1 },
			{ value: 'br', q: 1 },
			{ value: 'deflate', q: 0 },
		])
	})

	it('keeps a ;q=0 entry (an explicit rejection) rather than dropping it', () => {
		const entries = parseAcceptHeader('gzip;q=0, br')
		expect(entries).toContainEqual({ value: 'gzip', q: 0 })
		expect(entries[0]).toEqual({ value: 'br', q: 1 })
	})

	it('is a STABLE sort â€” equal-q entries keep their header order', () => {
		// br + gzip both at q=1 â†’ br stays first (the server preference a caller relies on for ties).
		expect(parseAcceptHeader('br, gzip, deflate').map((entry) => entry.value)).toEqual([
			'br',
			'gzip',
			'deflate',
		])
	})

	it('is total on an empty / whitespace header', () => {
		expect(parseAcceptHeader('')).toEqual([])
		expect(parseAcceptHeader('  ,  ')).toEqual([])
	})
})

describe('negotiateEncoding â€” coding selection from Accept-Encoding', () => {
	const offered = ['br', 'gzip', 'deflate'] as const

	it('picks the highest-q offered coding', () => {
		expect(negotiateEncoding('gzip;q=0.5, br;q=1.0', offered)).toBe('br')
	})

	it('honors the server order on a tie', () => {
		expect(negotiateEncoding('gzip, br', offered)).toBe('br')
	})

	it('honors a bare * wildcard (the first offered coding)', () => {
		expect(negotiateEncoding('*', offered)).toBe('br')
	})

	it('skips a ;q=0-rejected coding, even under a * wildcard', () => {
		expect(negotiateEncoding('br;q=0, *', offered)).toBe('gzip')
		expect(negotiateEncoding('br;q=0, gzip', offered)).toBe('gzip')
	})

	it('a named ;q=0 beats an EXPLICIT *;q=1 wildcard (the rejection wins, the chosen coding skips it)', () => {
		// `gzip;q=0` explicitly rejects gzip; the `*;q=1` would otherwise accept it. The specific
		// rejection beats the wildcard, so gzip is skipped and the wildcard lands on the next offered
		// coding (br, the server's first preference among the still-acceptable codings).
		expect(negotiateEncoding('gzip;q=0, *;q=1', ['gzip', 'br', 'deflate'])).toBe('br')
	})

	it('returns undefined when none of the offered codings is acceptable', () => {
		expect(negotiateEncoding('identity', offered)).toBeUndefined()
		expect(negotiateEncoding('', offered)).toBeUndefined()
	})
})

// codingQuality â€” the per-coding q lookup negotiateEncoding scores each offered coding with.
// The sharp edge: a NAMED `;q=0` (an explicit rejection) wins over a present `*` wildcard â€” the
// named match is consulted FIRST, so the rejection scores 0 even when `*;q=1` would otherwise apply.
describe('codingQuality â€” a specific rejection beats the wildcard', () => {
	it('scores a named ;q=0 as 0 even when a *;q=1 wildcard is present', () => {
		const entries = parseAcceptHeader('gzip;q=0, *;q=1')
		// The named `gzip;q=0` is found first â†’ 0 (rejected), NOT the wildcard's 1.
		expect(codingQuality(entries, 'gzip')).toBe(0)
		// An UN-named coding falls through to the `*;q=1` wildcard â†’ 1 (acceptable).
		expect(codingQuality(entries, 'br')).toBe(1)
	})

	it('scores an un-named coding as 0 when there is no wildcard to cover it', () => {
		const entries = parseAcceptHeader('gzip')
		expect(codingQuality(entries, 'gzip')).toBe(1)
		expect(codingQuality(entries, 'br')).toBe(0) // not named, no `*` â†’ not acceptable
	})
})

describe('compressBody / decompressBody â€” the node:zlib round-trip', () => {
	const payload = Buffer.from('the quick brown fox '.repeat(64))

	for (const coding of ['gzip', 'deflate', 'br'] as const) {
		it(`${coding} compresses smaller and decompresses back to the original`, () => {
			const compressed = compressBody(payload, coding)
			expect(compressed.length).toBeLessThan(payload.length) // a repetitive payload shrinks
			expect(decompressBody(compressed, coding).equals(payload)).toBe(true)
		})
	}

	it('produces a real gzip stream node:zlib itself can read', () => {
		// Cross-check: our gzip output decodes with node:zlib's own gunzipSync (a genuine gzip stream).
		expect(gunzipSync(compressBody(payload, 'gzip')).equals(payload)).toBe(true)
		expect(brotliDecompressSync(compressBody(payload, 'br')).equals(payload)).toBe(true)
	})

	it('decodes a stream node:zlib produced (the wire is the inverse both ways)', () => {
		expect(decompressBody(gzipSync(payload), 'gzip').equals(payload)).toBe(true)
	})

	it('identity is a pass-through (no transform)', () => {
		expect(compressBody(payload, 'identity').equals(payload)).toBe(true)
		expect(decompressBody(payload, 'identity').equals(payload)).toBe(true)
	})
})

describe('mergeVary â€” appending a Vary token without duplication', () => {
	it('returns the value alone when there is no existing Vary', () => {
		expect(mergeVary('', 'Accept-Encoding')).toBe('Accept-Encoding')
	})

	it('appends when the token is absent', () => {
		expect(mergeVary('Origin', 'Accept-Encoding')).toBe('Origin, Accept-Encoding')
	})

	it('is a no-op (case-insensitive) when the token is already present', () => {
		expect(mergeVary('accept-encoding', 'Accept-Encoding')).toBe('accept-encoding')
		expect(mergeVary('Origin, Accept-Encoding', 'Accept-Encoding')).toBe('Origin, Accept-Encoding')
	})

	it('appends a DISTINCT second token to an existing Accept-Encoding (both present, no dup)', () => {
		// Existing `Accept-Encoding`, add a distinct `Origin` â†’ both listed, comma-separated, once each.
		expect(mergeVary('Accept-Encoding', 'Origin')).toBe('Accept-Encoding, Origin')
	})

	it('does NOT duplicate a token that already exists in a DIFFERENT case', () => {
		// The existing token is upper-cased; appending its canonical-cased form must NOT re-add it
		// (the de-dupe lower-cases BOTH sides) â€” the existing value is returned verbatim.
		expect(mergeVary('ACCEPT-ENCODING', 'Accept-Encoding')).toBe('ACCEPT-ENCODING')
		expect(mergeVary('Origin, ACCEPT-ENCODING', 'accept-encoding')).toBe('Origin, ACCEPT-ENCODING')

describe('parsePartHeaders â€” part header parsing', () => {
	it('parses a FILE part (name + filename + content type)', () => {
		const raw = Buffer.from(
			'Content-Disposition: form-data; name="avatar"; filename="cat.png"\r\nContent-Type: image/png',
		)

			name: 'avatar',
			filename: 'cat.png',
			type: 'image/png',
		})
	})

	it('parses a FIELD part (no filename, no content type)', () => {
		const raw = Buffer.from('Content-Disposition: form-data; name="title"')
		expect(parsePartHeaders(raw)).toEqual({
			name: 'title',
			filename: undefined,
			type: undefined,
		})
	})

	it('preserves a traversal filename verbatim as metadata (the parser never builds a path from it)', () => {
		const raw = Buffer.from('Content-Disposition: form-data; name="f"; filename="../../etc/passwd"')
		expect(parsePartHeaders(raw).filename).toBe('../../etc/passwd')
	})

	it('reads an empty filename (the empty-string upload) as a file part', () => {
		const raw = Buffer.from('Content-Disposition: form-data; name="f"; filename=""')
		expect(parsePartHeaders(raw).filename).toBe('')
	})

	it('parses a quoted filename containing ; and = INTACT (the quote-bounded value)', () => {
		// `filename="a;b=c.png"` â€” the `;` and `=` live INSIDE the quotes; the filename matcher's
		// quote-bounded class (`[^"]*`) captures the whole thing rather than stopping at the `;`.
		const raw = Buffer.from(
			'Content-Disposition: form-data; name="f"; filename="a;b=c.png"\r\nContent-Type: image/png',
		)
		expect(parsePartHeaders(raw)).toEqual({ name: 'f', filename: 'a;b=c.png', type: 'image/png' })
	})

	it('a part with a Content-Type but NO filename is a FIELD (filename undefined)', () => {
		// A `Content-Type` line WITHOUT a `filename` parameter is still a form FIELD, not a file â€” the
		// PRESENCE of a filename (not the content type) is what marks a file part. `type` is read; the
		// part stays a field because `filename` is absent.
		const raw = Buffer.from(
			'Content-Disposition: form-data; name="meta"\r\nContent-Type: application/json',
		)
		expect(parsePartHeaders(raw)).toEqual({
			name: 'meta',
			filename: undefined,
			type: 'application/json',
		})
	})
})

describe('matchesMagic / detectMIME â€” magic-byte sniffing', () => {
	it('matches a signature with a null wildcard (the WebP RIFFã€ˆsizeã€‰WEBP marker)', () => {
		const webp = Buffer.from([
			0x52, 0x49, 0x46, 0x46, 0x10, 0x20, 0x30, 0x40, 0x57, 0x45, 0x42, 0x50,

			false,
		)
		// Too short to match.
		expect(matchesMagic(Buffer.from([0xff]), [0xff, 0xd8, 0xff])).toBe(false)
	})

	it('detects PNG / JPEG / GIF / PDF from leading bytes, and undefined for text', () => {
		expect(detectMIME(Buffer.from([0x89, 0x50, 0x4e, 0x47, 0x0d, 0x0a, 0x1a, 0x0a]))).toBe(
			'image/png',
		)
		expect(detectMIME(Buffer.from([0xff, 0xd8, 0xff, 0xe0]))).toBe('image/jpeg')
		expect(detectMIME(Buffer.from('GIF89a...'))).toBe('image/gif')
		expect(detectMIME(Buffer.from('%PDF-1.7'))).toBe('application/pdf')
		expect(detectMIME(Buffer.from('just plain text'))).toBeUndefined()
	})

	it('detectMIME returns undefined for a buffer SHORTER than every signature (the length guard)', () => {
		// The PNG prefix is the right bytes but only 3 of its 8 â€” too short for matchesMagic's length
		// guard, so detectMIME finds no match (a partial header is NOT sniffed as the full type).
		expect(detectMIME(Buffer.from([0x89, 0x50, 0x4e]))).toBeUndefined()
		// A single byte matches no signature's full length either.
		expect(detectMIME(Buffer.from([0xff]))).toBeUndefined()
		// The empty buffer is shorter than everything â†’ undefined (never throws).
		expect(detectMIME(Buffer.alloc(0))).toBeUndefined()
	})
})

describe('isMultipartBody â€” the parsed-body boundary guard', () => {
	it('accepts a { files, fields } record and rejects anything else', () => {
		expect(isMultipartBody({ files: [], fields: {} })).toBe(true)
		expect(isMultipartBody({ files: 'x', fields: {} })).toBe(false)
		expect(isMultipartBody({ files: [] })).toBe(false)
		expect(isMultipartBody('hello')).toBe(false)
		expect(isMultipartBody(undefined)).toBe(false)
	})
})

// The staged-upload operations are FREE FUNCTIONS over the frozen `UploadedFileInterface` record
// (no longer methods on a class â€” the value is plain, immutable data built by `createUploadedFile`).
// Exercised over REAL temp files (no mock FS â€” AGENTS Â§16) staged through `createStaticFixture`. The
// over-the-wire staging (a random temp name, the path-traversal defense) is proven through
// `createMultipart` in `middlewares.test.ts`; here each operation is unit-tested against a file we
// stage ourselves.
describe('readUploadedFile â€” read the staged bytes into a Buffer', () => {
	it('reads back the exact staged bytes', async () => {
		const bytes = Buffer.from([0x89, 0x50, 0x4e, 0x47, 0x01, 0x02, 0x03])
		const fixture = createStaticFixture({ blob: bytes })
		try {
			const file = createUploadedFile({
				field: 'f',
				name: 'x.bin',
				size: bytes.length,
				mime: 'application/octet-stream',
				validated: false,
				path: join(fixture.root, 'blob'),
			})
			const read = await readUploadedFile(file)
			expect(Buffer.from(read).equals(bytes)).toBe(true)
		} finally {
			fixture.cleanup()
		}
	})
})

describe('streamUploadedFile â€” stream the staged bytes', () => {
	it('streams the exact staged bytes', async () => {
		const bytes = Buffer.from('the quick brown fox')
		const fixture = createStaticFixture({ blob: bytes })
		try {
			const file = createUploadedFile({
				field: 'f',
				name: 'x.txt',
				size: bytes.length,
				mime: 'text/plain',
				validated: false,
				path: join(fixture.root, 'blob'),
			})
			const chunks: Buffer[] = []
			for await (const chunk of streamUploadedFile(file)) {
				chunks.push(Buffer.isBuffer(chunk) ? chunk : Buffer.from(String(chunk)))
			}
			expect(Buffer.concat(chunks).equals(bytes)).toBe(true)
		} finally {
			fixture.cleanup()
		}
	})
})

describe('moveUploadedFile â€” relocate, returning a NEW frozen record', () => {
	it('moves the bytes and returns a new record (status `moved`, repointed path) without mutating the original', async () => {
		const bytes = Buffer.from('payload')
		const fixture = createStaticFixture({ blob: bytes })
		try {
			const source = join(fixture.root, 'blob')
			const destination = join(fixture.root, 'final.dat')
			const file = createUploadedFile({
				field: 'f',
				name: 'x.dat',
				size: bytes.length,
				mime: 'application/octet-stream',
				validated: false,
				path: source,
			})
			const moved = await moveUploadedFile(file, destination)
			// The returned record carries the new path + `moved` status; the ORIGINAL is unchanged (the
			// record is frozen / immutable â€” the move never mutates it in place).
			expect(moved.status).toBe('moved')
			expect(moved.path).toBe(destination)
			expect(file.status).toBe('staged')
			expect(file.path).toBe(source)
			// The destination now holds the bytes; reading through the returned record works (off the new
			// path). (The EXDEV cross-device fallback â€” a copy + unlink when `rename` fails across
			// filesystems â€” is not reproducible on a single-device CI temp dir, so it is documented +
			// covered by the structural `'code' in error && error.code === 'EXDEV'` guard in the helper;
			// a same-device rename is the path exercised here.)
			expect(readFileSync(destination).equals(bytes)).toBe(true)
			expect(Buffer.from(await readUploadedFile(moved)).equals(bytes)).toBe(true)
		} finally {
			fixture.cleanup()
		}
	})

	it('surfaces a real rename fault (a non-existent destination directory)', async () => {
		const fixture = createStaticFixture({ blob: Buffer.from('x') })
		try {
			const source = join(fixture.root, 'blob')
			const file = createUploadedFile({
				field: 'f',
				name: 'x.dat',
				size: 1,
				mime: 'application/octet-stream',
				validated: false,
				path: source,
			})
			await expect(
				moveUploadedFile(file, join(source, 'no', 'such', 'dir', 'out.dat')),
			).rejects.toBeInstanceOf(Error)
		} finally {
			fixture.cleanup()
		}
	})
})

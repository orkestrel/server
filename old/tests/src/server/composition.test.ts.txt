import type { StartedServerInterface, StaticFixtureInterface } from '../../../setupServer.js'
import { gunzipSync } from 'node:zlib'
import { describe, expect, it } from 'vitest'
import {
	createBodyParser,
	createCompression,
	createCookieSession,
	createCors,
	createCSRFGuard,
	createErrorBoundary,
	createETag,
	createRateLimiter,
	createSecurity,
	createServer,
	createStatic,
	CSRF_STATE,
	HTTPError,
} from '@src/server'
import {
	cookiesHeader,
	createStaticFixture,
	createTeardown,
	readRawResponse,
	readSetCookie,
	startServer,
} from '../../../setupServer.js'

// src/server/http — the S-g CAPSTONE: a full production-shaped stack composed on ONE real
// `node:http` server, proving every battery middleware shipped across S-a…S-f composes end to
// end in the recommended order. NO mocks (AGENTS §16): a real Server on an ephemeral port,
// driven by the global `fetch` (which exposes Set-Cookie / headers but transparently decompresses)
// AND by `readRawResponse` (raw node:http — does NOT auto-decompress, so the gzip-on-the-wire +
// 304-through-compression proofs read the real compressed bytes). The recommended order under test
// (outer → inner):
//
//   createCompression()           — OUTERMOST: compresses EVERYTHING downstream, incl. error bodies
//     createErrorBoundary()       — catches a downstream throw → status (its body is still compressed)
//       createSecurity()          — secure-default headers + X-Request-ID, early so every response carries them
//         createETag()            — content ETag over the UNCOMPRESSED body (inner of compression) + 304
//           createCors()          — cross-origin headers / preflight, early
//             createRateLimiter() — per-client window (well above the test's request count)
//               createBodyParser()        — parse + size-limit (so a form-field CSRF token is readable)
//                 createCookieSession()   — signed-cookie session id (emits its Set-Cookie)
//                   createCSRFGuard()     — double-submit CSRF (emits ITS Set-Cookie; mutating → verify)
//                     createStatic()      — serve a dir, FALL THROUGH (next()) to the routes on no match
//                       routes            — the application handlers
//
// The two Set-Cookie survival (session + csrf) is the S-c multi-Set-Cookie-through-compression fix;
// the ETag-inner-of-compression + error-body-compressed orderings are the S-e/S-c constraints.

const { track } = createTeardown((handle: StartedServerInterface) => handle.stop())
const fixtures = createTeardown((fixture: StaticFixtureInterface) => fixture.cleanup())

// A large, repetitive JSON payload that clears the default 1 KiB compression threshold and
// compresses well — so a gzip body is provably smaller than the original on the wire.
const PAYLOAD = { rows: Array.from({ length: 300 }, (_value, index) => `row-${index}`) }
const PAYLOAD_JSON = JSON.stringify(PAYLOAD)

const SESSION_SECRET = 'session-secret'
const CSRF_SECRET = 'csrf-secret'

// A large client-facing error message (well over the 1 KiB compression threshold) so the
// error-boundary's response body genuinely COMPRESSES — proving compression is OUTERMOST (a tiny
// generic 500 body would sit under the threshold and pass through, hiding the ordering proof).
const BOOM_MESSAGE = 'boom '.repeat(400).trim()

// Stand the full production stack in the recommended order, then register the test's routes +
// a static fixture. One builder so every test composes the IDENTICAL stack (the thing under proof).
function fullStack(root: string): StartedServerInterface['server'] {
	const server = createServer()
	server.use(createCompression()) // 1. OUTERMOST — compresses everything (incl. error bodies)
	server.use(createErrorBoundary()) // 2. catches downstream throws → status (still compressed)
	server.use(createSecurity()) // 3. secure-default headers + X-Request-ID, early
	server.use(createETag()) // 4. content ETag over the UNCOMPRESSED body, inner of compression
	server.use(createCors()) // 5. cross-origin headers / preflight, early
	server.use(createRateLimiter({ max: 1_000, window: 60_000 })) // 6. per-client window (well above the test)
	server.use(createBodyParser()) // 7. parse + size-limit (a form-field CSRF token is readable)
	server.use(createCookieSession({ secret: SESSION_SECRET })) // 8. signed-cookie session (emits Set-Cookie)
	server.use(createCSRFGuard({ secret: CSRF_SECRET })) // 9. double-submit CSRF (emits Set-Cookie; verifies)
	server.use(createStatic({ root, maxAge: 3600 })) // 10. serve the dir, fall through (next()) to routes
	// A safe GET JSON route: large compressible body + exposes the CSRF token for the page to echo.
	server.route({
		method: 'GET',
		path: '/data',
		handler: (context) => context.json({ ...PAYLOAD, csrf: context.state.get(CSRF_STATE) }),
	})
	// A safe GET with a BYTE-STABLE large body (no per-request CSRF token in it) — so its content ETag
	// is identical across requests and an `If-None-Match` revalidation can match. (The `/data` body
	// embeds a fresh random CSRF token each call, so its tag deliberately changes.)
	server.route({ method: 'GET', path: '/stable', handler: (context) => context.json(PAYLOAD) })
	// A safe GET that throws — the error boundary must catch it into a status response (and that body,
	// being INNER of compression, still compresses). A large `HTTPError` message clears the threshold so
	// the compression is observable on the wire.
	server.route({
		method: 'GET',
		path: '/boom',
		handler: () => {
			throw new HTTPError(500, BOOM_MESSAGE)
		},
	})
	// A mutating POST guarded by CSRF — rejected (403) without a token, accepted with one.
	server.route({ method: 'POST', path: '/data', handler: (context) => context.text('mutated') })
	return server
}

// Mint a CSRF token + its signed cookie via a safe GET to the stack — both needed to drive the
// mutating-request assertions (the double-submit pair the client echoes).
async function mintCSRF(
	handle: StartedServerInterface,
): Promise<{ readonly token: string; readonly cookie: string; readonly session: string }> {
	const response = await fetch(`${handle.base}/data`)
	const token = (await response.json()).csrf
	const cookie = readSetCookie(response, 'csrf')
	// The CSRF token is SESSION-BOUND (the cookie session precedes the guard in this stack), so a
	// mutating request must carry the SAME session it was minted under — capture the session cookie too
	// (a real browser sends both), else the POST mints a fresh session and the bound token mismatches.
	const session = readSetCookie(response, 'session')
	if (typeof token !== 'string' || cookie === undefined || session === undefined)
		throw new Error('csrf mint failed')
	return { token, cookie, session }
}

describe('the full server stack composes on one server (S-g capstone)', () => {
	// A real on-disk static fixture served IN FRONT of the routes — the stack must fall through it
	// (next() on no match) to reach the JSON / CSRF routes, and serve the asset when it DOES match.
	function freshFixture(): StaticFixtureInterface {
		return fixtures.track(createStaticFixture({ 'app.js': 'console.log("static")\n' }))
	}

	it('a JSON route carries gzip + security + ETag + BOTH cookies + CORS all at once', async () => {
		const handle = track(await startServer(fullStack(freshFixture().root)))

		// `fetch` exposes Set-Cookie / security / ETag / CORS (it transparently decompresses, so it
		// can't observe gzip — the raw read below proves that). Send an Origin so CORS echoes.
		const response = await fetch(`${handle.base}/data`, {
			headers: { 'accept-encoding': 'gzip', origin: 'https://app.example' },
		})
		expect(response.status).toBe(200)

		// Security headers (createSecurity) — the nosniff + frame + request-id correlation id.
		expect(response.headers.get('x-content-type-options')).toBe('nosniff')
		expect(response.headers.get('x-frame-options')).toBe('DENY')
		expect(response.headers.get('x-request-id')).toBeTruthy()

		// A content ETag (createETag over the uncompressed body) — weak by default.
		expect(response.headers.get('etag')).toMatch(/^W\/"/)

		// CORS (createCors) — the wildcard allow-origin is set.
		expect(response.headers.get('access-control-allow-origin')).toBe('*')

		// BOTH cookies survive (the S-c multi-Set-Cookie-through-compression fix): one session, one csrf.
		const cookies = response.headers.getSetCookie()
		expect(cookies.some((entry) => entry.startsWith('session='))).toBe(true)
		expect(cookies.some((entry) => entry.startsWith('csrf='))).toBe(true)

		// The body is intact (fetch decoded it) — the route's rows round-trip.
		expect((await response.json()).rows).toEqual(PAYLOAD.rows)

		// Now PROVE the wire was actually gzip-compressed (raw node:http does NOT auto-decompress) —
		// the compressed bytes are smaller than the JSON and gunzip back to a superset of the payload.
		const raw = await readRawResponse(handle.base, '/data', { 'accept-encoding': 'gzip' })
		expect(raw.headers['content-encoding']).toBe('gzip')
		expect(raw.body.length).toBeLessThan(Buffer.byteLength(PAYLOAD_JSON))
		expect(gunzipSync(raw.body).toString()).toContain('"row-0"')
		// Both Set-Cookies survive on the WIRE too (parsed as a string[], not comma-folded into one).
		const wireCookies = raw.headers['set-cookie'] ?? []
		expect(wireCookies.length).toBe(2)
	})

	it('a thrown handler error is caught by the boundary AND still compressed', async () => {
		const handle = track(await startServer(fullStack(freshFixture().root)))

		// The compression is OUTERMOST, so even the error-boundary's 500 body is gzip on the wire —
		// proving the order (a boundary OUTSIDE compression would leave the error body uncompressed).
		const raw = await readRawResponse(handle.base, '/boom', { 'accept-encoding': 'gzip' })
		expect(raw.status).toBe(500)
		expect(raw.headers['content-encoding']).toBe('gzip')
		// The compressed 500 body decodes to the boundary's JSON error shape (not a crashed socket).
		expect(JSON.parse(gunzipSync(raw.body).toString())).toEqual({ error: BOOM_MESSAGE })
	})

	it('a CSRF-protected POST is rejected (403) without a token and accepted with one', async () => {
		const handle = track(await startServer(fullStack(freshFixture().root)))
		const { token, cookie, session } = await mintCSRF(handle)
		// The realistic browser carries BOTH cookies (the session id + the CSRF token) on every request;
		// the CSRF token is session-bound, so the session cookie must ride along or the POST mints a new
		// session whose id the bound token doesn't match.
		const cookies = cookiesHeader({ session, csrf: cookie })

		// No token → the CSRF guard rejects with a 403 the error boundary renders.
		const rejected = await fetch(`${handle.base}/data`, {
			method: 'POST',
			headers: cookies,
		})
		expect(rejected.status).toBe(403)
		expect(await rejected.json()).toEqual({ error: 'invalid csrf token' })

		// The echoed token (header) matching the signed cookie (on the SAME session) → the mutation
		// passes through the stack.
		const accepted = await fetch(`${handle.base}/data`, {
			method: 'POST',
			headers: { ...cookies, 'x-csrf-token': token },
		})
		expect(accepted.status).toBe(200)
		expect(await accepted.text()).toBe('mutated')
	})

	it('a static asset is served through the same stack (the mount falls through to it)', async () => {
		const handle = track(await startServer(fullStack(freshFixture().root)))

		// createStatic sits in front of the routes; an existing file is served by it (with its MIME +
		// Cache-Control), while the security headers from the outer middleware still ride along.
		const response = await fetch(`${handle.base}/app.js`)
		expect(response.status).toBe(200)
		expect(response.headers.get('content-type')).toBe('text/javascript; charset=utf-8')
		expect(response.headers.get('cache-control')).toBe('max-age=3600')
		expect(response.headers.get('x-content-type-options')).toBe('nosniff') // the outer security cap
		expect(await response.text()).toBe('console.log("static")\n')
	})

	it('a conditional If-None-Match revalidation 304s through compression', async () => {
		const handle = track(await startServer(fullStack(freshFixture().root)))

		// First GET yields the ETag (computed inner of compression, over the uncompressed body). Use the
		// byte-stable route so the tag is reproducible (the `/data` body's random CSRF token would change it).
		const first = await readRawResponse(handle.base, '/stable', { 'accept-encoding': 'gzip' })
		const etag = first.headers.etag
		expect(typeof etag).toBe('string')
		expect(first.headers['content-encoding']).toBe('gzip')

		// Re-request with that ETag → a 304 (no body) DESPITE the compression sitting outside the ETag —
		// the validator is the uncompressed-content hash, so revalidation works regardless of coding.
		const revalidated = await readRawResponse(handle.base, '/stable', {
			'accept-encoding': 'gzip',
			'if-none-match': typeof etag === 'string' ? etag : '',
		})
		expect(revalidated.status).toBe(304)
		expect(revalidated.body.length).toBe(0)
	})
})

import type { StartedServerInterface } from '../../../setupServer.js'
import type { IncomingMessage } from 'node:http'
import type { Duplex } from 'node:stream'
import { describe, expect, it } from 'vitest'
import { createServer, discoverPort } from '@src/server'
import { createErrorRecorder, createRecorder, waitForDelay } from '../../../setup.js'
import { createTeardown, startServer, upgradeRequest } from '../../../setupServer.js'

// src/server/http/Server.ts — the lifecycle facade over REAL node:http (no mocks,
// the node src:server project). Each test stands a server on an ephemeral port,
// drives it with real fetch, and stops + destroys it in afterEach so no socket
// leaks. Covers: lifecycle + status transitions, routing (params/query/404/405/
// HEAD/OPTIONS/groups), graceful drain (an in-flight slow handler completes; new
// requests refused after stop), the lifecycle events, emit-safety (a throwing
// observer can't crash the server, routes to the emitter's error handler), and the upgrade seam
// (a raw node:http upgrade fans out to handlers — first-claimer-wins owns the
// socket, an unclaimed upgrade is destroyed, and the `upgrade` event reports it).

const { track } = createTeardown((handle: StartedServerInterface) => handle.stop())

describe('Server — lifecycle', () => {
	it('starts on an ephemeral port, serves a route, and reports listening', async () => {
		const server = createServer()
		server.route({ method: 'GET', path: '/ping', handler: (c) => c.text('pong') })
		expect(server.status).toBe('idle')
		const handle = track(await startServer(server))
		expect(server.status).toBe('listening')
		expect(server.port).toBeGreaterThan(0)
		expect(server.id).toMatch(/[0-9a-f-]{36}/)
		const response = await fetch(`${handle.base}/ping`)
		expect(await response.text()).toBe('pong')
	})

	it('transitions idle → listening → stopped and refuses start while listening', async () => {
		const server = createServer()
		track(await startServer(server))
		await expect(server.start()).rejects.toThrow(/cannot start/)
		expect(server.status).toBe('listening')
	})

	it('destroy is idempotent and leaves the server stopped', async () => {
		const server = createServer()
		await server.start()
		await server.destroy()
		await server.destroy()
		expect(server.status).toBe('stopped')
		expect(server.port).toBeUndefined()
	})

	it('a thrown handler error becomes a 500 (the server never crashes)', async () => {
		const server = createServer()
		server.route({
			method: 'GET',
			path: '/boom',
			handler: () => {
				throw new Error('kaboom')
			},
		})
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/boom`)
		expect(response.status).toBe(500)
		// The server is still up — a later request still works.
		server.route({ method: 'GET', path: '/ok', handler: (c) => c.text('ok') })
		expect(await (await fetch(`${handle.base}/ok`)).text()).toBe('ok')
	})
})

describe('Server — host/port bind', () => {
	it('binds the exact configured port (and the port member reflects it)', async () => {
		// Discover a free port, then bind THAT exact port — start() must resolve it
		// unchanged (not an ephemeral substitute), and the `port` member reflects it.
		const fixed = await discoverPort()
		const server = createServer({ port: fixed })
		server.route({ method: 'GET', path: '/ping', handler: (c) => c.text('pong') })
		const handle = track(await startServer(server))
		expect(handle.port).toBe(fixed)
		expect(server.port).toBe(fixed)
		// The fixed port is genuinely serving on the loopback URL.
		expect(await (await fetch(`http://127.0.0.1:${fixed}/ping`)).text()).toBe('pong')
	})

	it('binds the configured host and is reachable on it', async () => {
		// Bind loopback explicitly with a discovered free port — the server must be
		// reachable on 127.0.0.1 (proving the host argument reached `server.listen`).
		const fixed = await discoverPort()
		const server = createServer({ host: '127.0.0.1', port: fixed })
		server.route({ method: 'GET', path: '/where', handler: (c) => c.text('here') })
		const handle = track(await startServer(server))
		expect(handle.port).toBe(fixed)
		const response = await fetch(`http://127.0.0.1:${fixed}/where`)
		expect(await response.text()).toBe('here')
	})

	it('defaults to an ephemeral port when none is configured (the shipped behavior)', async () => {
		// No `port` ⇒ the OS assigns a free one (the existing default every shipped test
		// relies on). It resolves to a real, non-zero, reachable port.
		const server = createServer()
		server.route({ method: 'GET', path: '/p', handler: (c) => c.text('p') })
		const handle = track(await startServer(server))
		expect(handle.port).toBeGreaterThan(0)
		expect(server.port).toBe(handle.port)
		expect(await (await fetch(`${handle.base}/p`)).text()).toBe('p')
	})

	it('rejects start() when the configured port is already in use', async () => {
		// Stand a first server on a discovered port, then a SECOND on the SAME fixed port —
		// the OS refuses the duplicate bind, so start() rejects with EADDRINUSE (it does not
		// silently fall back to an ephemeral port — a fixed bind is honored or it fails).
		const fixed = await discoverPort()
		const first = createServer({ port: fixed })
		track(await startServer(first))
		const second = createServer({ port: fixed })
		await expect(second.start()).rejects.toThrow(/EADDRINUSE/)
		// The failed start left the second server back at idle (the onError reset path).
		expect(second.status).toBe('idle')
	})
})

describe('Server — restart + idempotent lifecycle', () => {
	it('restarts after a stop and serves again (a restarted server is not born aborted)', async () => {
		// start → stop → start AGAIN: the second run must mint a FRESH stop signal
		// (Server.ts ~121) so the new run is NOT born aborted — a route serves 200.
		const server = createServer()
		server.route({ method: 'GET', path: '/ping', handler: (c) => c.text('pong') })
		// First run.
		const first = await startServer(server)
		expect((await fetch(`${first.base}/ping`)).status).toBe(200)
		await server.stop()
		expect(server.status).toBe('stopped')
		// Second run — a NEW ephemeral port; track this handle for afterEach cleanup.
		const second = track(await startServer(server))
		expect(server.status).toBe('listening')
		expect(server.port).toBeGreaterThan(0)
		const response = await fetch(`${second.base}/ping`)
		expect(response.status).toBe(200)
		expect(await response.text()).toBe('pong')
	})

	it('stop() before start() is a safe no-op (stays idle, no throw)', async () => {
		// stop() on an idle server returns early (status !== 'listening') — no throw,
		// the status stays idle, and a later start() still works.
		const server = createServer()
		expect(server.status).toBe('idle')
		await expect(server.stop()).resolves.toBeUndefined()
		expect(server.status).toBe('idle')
		// The no-op stop did not poison the server — it can still start + serve.
		server.route({ method: 'GET', path: '/ok', handler: (c) => c.text('ok') })
		const handle = track(await startServer(server))
		expect(await (await fetch(`${handle.base}/ok`)).text()).toBe('ok')
	})

	it('double stop() is a no-op — the second call does not throw or change state', async () => {
		// After the first stop() the status is 'stopped'; a second stop() hits the same
		// `status !== 'listening'` guard and returns immediately.
		const server = createServer()
		await startServer(server)
		await server.stop()
		expect(server.status).toBe('stopped')
		await expect(server.stop()).resolves.toBeUndefined()
		expect(server.status).toBe('stopped')
		expect(server.port).toBeUndefined()
	})

	it('destroy() while LISTENING tears down from any state (no prior stop)', async () => {
		// destroy() called straight from 'listening' (no stop() first) must abort + close
		// + destroy the emitter, landing 'stopped' — and stay idempotent on a repeat.
		const server = createServer()
		server.route({ method: 'GET', path: '/x', handler: (c) => c.text('x') })
		const handle = await startServer(server)
		expect(server.status).toBe('listening')
		await server.destroy()
		expect(server.status).toBe('stopped')
		expect(server.port).toBeUndefined()
		// Idempotent — a second destroy from 'stopped' is still a clean no-op.
		await expect(server.destroy()).resolves.toBeUndefined()
		expect(server.status).toBe('stopped')
		// The socket is genuinely closed — a new connection is refused.
		await expect(fetch(`${handle.base}/x`)).rejects.toBeDefined()
	})
})

describe('Server — concurrency', () => {
	it('serves N parallel requests against a slow handler — none dropped, every body correct', async () => {
		// Fire 20 concurrent requests at a slow-ish handler that echoes its id. Real
		// parallelism drives the RequestManager count up then back to zero; every
		// response must carry the right body (the drain bookkeeping drops nothing).
		const server = createServer()
		server.route({
			method: 'GET',
			path: '/echo/:id',
			handler: async (c) => {
				await waitForDelay(40)
				c.text(c.params.id)
			},
		})
		const handle = track(await startServer(server))
		const ids = Array.from({ length: 20 }, (_value, index) => String(index))
		const bodies = await Promise.all(
			ids.map(async (id) => (await fetch(`${handle.base}/echo/${id}`)).text()),
		)
		expect(bodies.sort((a, b) => Number(a) - Number(b))).toEqual(ids)
	})
})

describe('Server — discoverPort pre-pick flow', () => {
	it('pre-picks a free port via discoverPort, binds it, and serves on exactly that port', async () => {

	})
})

describe('Server — routing', () => {
	it('matches path params and parses the query', async () => {
		const server = createServer()
		server.route({
			method: 'GET',
			path: '/users/:id',
			handler: (c) => c.json({ id: c.params.id, sort: c.query.get('sort') }),
		})
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/users/7?sort=name`)
		expect(await response.json()).toEqual({ id: '7', sort: 'name' })
	})

	it('answers 404 for an unknown path', async () => {
		const server = createServer()
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/missing`)
		expect(response.status).toBe(404)
		expect(await response.json()).toEqual({ error: 'Not Found' })
	})

	it('answers 405 with the Allow header for a wrong method', async () => {
		const server = createServer()
		server.route({ method: 'GET', path: '/r', handler: (c) => c.text('r') })
		server.route({ method: 'POST', path: '/r', handler: (c) => c.text('r') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/r`, { method: 'DELETE' })
		expect(response.status).toBe(405)
		const allow = (response.headers.get('allow') ?? '').split(', ').sort()
		expect(allow).toEqual(['GET', 'HEAD', 'POST'])
	})

	it('auto-answers HEAD from a GET route (no body)', async () => {
		const server = createServer()
		server.route({ method: 'GET', path: '/page', handler: (c) => c.text('hello') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/page`, { method: 'HEAD' })
		expect(response.status).toBe(200)
		expect(await response.text()).toBe('')
	})

	it('answers OPTIONS with the Allow set', async () => {
		const server = createServer()
		server.route({ method: 'GET', path: '/r', handler: (c) => c.text('r') })
		server.route({ method: 'PUT', path: '/r', handler: (c) => c.text('r') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/r`, { method: 'OPTIONS' })
		expect(response.status).toBe(204)
		const allow = (response.headers.get('allow') ?? '').split(', ').sort()
		expect(allow).toEqual(['GET', 'HEAD', 'OPTIONS', 'PUT'])
	})

	it('answers 404 (NOT 204) for OPTIONS on an unknown path', async () => {
		// The dispatch answers OPTIONS only when the path HAS methods (allow.length > 0);
		// an unknown path has none, so OPTIONS falls through to the 404 branch — it is NOT
		// a blanket 204. (Contrast with the happy OPTIONS test above, which is 204.)
		const server = createServer()
		server.route({ method: 'GET', path: '/known', handler: (c) => c.text('k') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/unknown`, { method: 'OPTIONS' })
		expect(response.status).toBe(404)
		expect(await response.json()).toEqual({ error: 'Not Found' })
	})

	it('answers 405 with Allow for HEAD on a POST-only route (HEAD does not satisfy POST)', async () => {
		// HEAD is implicitly served only by a GET route; a POST-only path does NOT answer
		// HEAD, so it is 405 (unmethoded) carrying the Allow set. The POST-only path has no
		// GET, so no implicit HEAD is advertised — Allow is exactly ['POST'].
		const server = createServer()
		server.route({ method: 'POST', path: '/only-post', handler: (c) => c.text('p') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/only-post`, { method: 'HEAD' })
		expect(response.status).toBe(405)
		const allow = (response.headers.get('allow') ?? '').split(', ').sort()
		expect(allow).toEqual(['POST'])
	})

	it('allows HEAD on a GET route (the positive contrast to HEAD on POST-only)', async () => {
		// The contrast asserted alongside the 405 case: HEAD on a GET route IS served
		// (the GET handler runs, the body is discarded) — proving the 405 above is about
		// the POST-only method set, not a blanket HEAD refusal.
		const server = createServer()
		server.route({ method: 'GET', path: '/get-only', handler: (c) => c.text('body') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/get-only`, { method: 'HEAD' })
		expect(response.status).toBe(200)
		expect(await response.text()).toBe('')
	})

	it('a re-registered method+path is last-write-wins (the second handler serves)', async () => {
		// Registering the same method+path twice via server.route(...) REPLACES the first
		// (RouteManager.add last-write-wins) — the second handler is the one that serves.
		const server = createServer()
		server.route({ method: 'GET', path: '/dup', handler: (c) => c.text('first') })
		server.route({ method: 'GET', path: '/dup', handler: (c) => c.text('second') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/dup`)
		expect(response.status).toBe(200)
		expect(await response.text()).toBe('second')
	})

	it('serves routes registered under a group prefix', async () => {
		const server = createServer()
		const api = server.group('/api')
		api.add({ method: 'GET', path: '/health', handler: (c) => c.json({ ok: true }) })
		const handle = track(await startServer(server))
		expect((await fetch(`${handle.base}/api/health`)).status).toBe(200)
		expect((await fetch(`${handle.base}/health`)).status).toBe(404)
	})
})

describe('Server — graceful drain', () => {
	it('lets an in-flight slow handler finish, then refuses new requests', async () => {
		const server = createServer({ drain: 2_000 })
		server.route({
			method: 'GET',
			path: '/slow',
			handler: async (c) => {
				await waitForDelay(120)
				c.text('done')
			},
		})
		const handle = track(await startServer(server))
		const base = handle.base
		// Fire a slow request, let it reach the handler, THEN stop — it must complete.
		const inflight = fetch(`${base}/slow`)
		await waitForDelay(30)
		const stopping = server.stop()
		const response = await inflight
		expect(await response.text()).toBe('done')
		await stopping
		expect(server.status).toBe('stopped')
		// A new connection after stop is refused (the socket is closed).
		await expect(fetch(`${base}/slow`)).rejects.toBeDefined()
	})

	it('forces closed after the drain deadline when a handler hangs', async () => {
		const server = createServer({ drain: 40 })
		server.route({
			method: 'GET',
			path: '/hang',
			handler: async () => {
				// Never responds — the drain deadline must fire and force-close.
				await waitForDelay(10_000)
			},
		})
		const handle = track(await startServer(server))
		const inflight = fetch(`${handle.base}/hang`).catch((error: unknown) => error)
		await waitForDelay(20)
		// stop() resolves once the drain deadline fires, even though the handler hangs.
		await server.stop()
		expect(server.status).toBe('stopped')
		await inflight
	})
})

describe('Server — lifecycle events (§13)', () => {
	it('fires start / request / stop / drain through the emitter', async () => {
		const starts = createRecorder<readonly [number]>()
		const requests = createRecorder<readonly [string, string]>()
		const stops = createRecorder<readonly []>()
		const drains = createRecorder<readonly [number]>()
		const server = createServer({
			on: { start: starts.handler, stop: stops.handler, drain: drains.handler },
		})
		server.emitter.on('request', requests.handler)
		server.route({ method: 'GET', path: '/x', handler: (c) => c.text('x') })
		const handle = track(await startServer(server))
		await fetch(`${handle.base}/x`)
		await server.stop()
		expect(starts.count).toBe(1)
		expect(starts.calls[0]?.[0]).toBe(handle.port)
		expect(requests.calls).toContainEqual(['GET', '/x'])
		expect(stops.count).toBe(1)
		expect(drains.count).toBe(1)
	})

	it('fires an error event when a handler throws', async () => {
		const errors = createRecorder<readonly [unknown]>()
		const server = createServer({ on: { error: errors.handler } })
		server.route({
			method: 'GET',
			path: '/boom',
			handler: () => {
				throw new Error('observed')
			},
		})
		const handle = track(await startServer(server))
		await fetch(`${handle.base}/boom`)
		expect(errors.count).toBe(1)
	})
})

describe('Server — emit safety', () => {
	it('a throwing observer cannot crash the server and routes to the error handler', async () => {
		const errors = createErrorRecorder()
		const server = createServer({ error: errors.handler })
		// A request observer that always throws — it must NOT prevent the response.
		server.emitter.on('request', () => {
			throw new Error('bad observer')
		})
		server.route({ method: 'GET', path: '/ok', handler: (c) => c.text('still works') })
		const handle = track(await startServer(server))
		const response = await fetch(`${handle.base}/ok`)
		expect(await response.text()).toBe('still works')
		// The isolated observer throw routed to the emitter's error handler (error, event).
		expect(errors.count).toBeGreaterThanOrEqual(1)
		expect(errors.calls[0]?.[1]).toBe('request')
	})
})

describe('Server — upgrade seam', () => {
	it('fans a raw upgrade out to a handler that claims the socket, and fires upgrade(handled: true)', async () => {
		const upgrades = createRecorder<readonly [IncomingMessage, boolean]>()
		// Capture the raw triple the handler receives, then CLAIM the socket — answer the
		// 101 handshake and end the connection (return true = "I own this socket").
		let seen: { request: IncomingMessage; socket: Duplex; head: Buffer } | undefined
		const server = createServer()
		server.emitter.on('upgrade', upgrades.handler)
		server.upgrade((request, socket, head) => {
			seen = { request, socket, head }
			socket.write(
				'HTTP/1.1 101 Switching Protocols\r\nUpgrade: websocket\r\nConnection: Upgrade\r\n\r\n',
			)
			socket.end()
			return true
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/ws', { 'Sec-WebSocket-Key': 'abc123' })
		// The client saw the 101 — a handler claimed the socket.
		expect(outcome.claimed).toBe(true)
		expect(outcome.status).toBe(101)
		// The handler received the real upgrade request + a live Duplex socket + a Buffer head.
		expect(seen?.request.url).toBe('/ws')
		expect(seen?.request.headers.upgrade).toBe('websocket')
		expect(seen?.request.headers['sec-websocket-key']).toBe('abc123')
		expect(typeof seen?.socket.write).toBe('function')
		expect(typeof seen?.socket.end).toBe('function')
		expect(Buffer.isBuffer(seen?.head)).toBe(true)
		// The observability event fired with handled: true, carrying the upgrade request.
		expect(upgrades.count).toBe(1)
		expect(upgrades.calls[0]?.[0].url).toBe('/ws')
		expect(upgrades.calls[0]?.[1]).toBe(true)
	})

	it('destroys the socket and fires upgrade(handled: false) when no handler claims it', async () => {
		const upgrades = createRecorder<readonly [IncomingMessage, boolean]>()
		const server = createServer()
		server.emitter.on('upgrade', upgrades.handler)
		// A handler that DECLINES (returns false) — the only registered one, so nothing claims.
		let declinedSocket: Duplex | undefined
		server.upgrade((_request, socket) => {
			declinedSocket = socket
			return false
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/nope')
		// No 101 reached the client — the spine destroyed the un-upgraded socket.
		expect(outcome.claimed).toBe(false)
		expect(declinedSocket?.destroyed).toBe(true)
		// The event reported the unclaimed upgrade.
		expect(upgrades.count).toBe(1)
		expect(upgrades.calls[0]?.[1]).toBe(false)
	})

	it('destroys the socket when there are NO upgrade handlers at all', async () => {
		const upgrades = createRecorder<readonly [IncomingMessage, boolean]>()
		const server = createServer()
		server.emitter.on('upgrade', upgrades.handler)
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base)
		expect(outcome.claimed).toBe(false)
		expect(upgrades.count).toBe(1)
		expect(upgrades.calls[0]?.[1]).toBe(false)
	})

	it('first claimer wins — a later handler never runs', async () => {
		const order: string[] = []
		const server = createServer()
		// First handler claims; the second must NEVER be consulted.
		server.upgrade((_request, socket) => {
			order.push('first')
			socket.write(
				'HTTP/1.1 101 Switching Protocols\r\nUpgrade: websocket\r\nConnection: Upgrade\r\n\r\n',
			)
			socket.end()
			return true
		})
		server.upgrade(() => {
			order.push('second')
			return true
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/ws')
		expect(outcome.claimed).toBe(true)
		expect(order).toEqual(['first'])
	})

	it('skips a declining handler and lets the next one claim', async () => {
		const order: string[] = []
		const server = createServer()
		// The first declines (false); the second claims — the fan-out must reach it.
		server.upgrade(() => {
			order.push('decline')
			return false
		})
		server.upgrade((_request, socket) => {
			order.push('claim')
			socket.write(
				'HTTP/1.1 101 Switching Protocols\r\nUpgrade: websocket\r\nConnection: Upgrade\r\n\r\n',
			)
			socket.end()
			return true
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/ws')
		expect(outcome.claimed).toBe(true)
		expect(order).toEqual(['decline', 'claim'])
	})

	it('a throwing handler is treated as declined — the next one still claims and the server survives', async () => {
		const order: string[] = []
		const errors = createRecorder<readonly [unknown]>()
		const server = createServer()
		server.emitter.on('error', errors.handler)
		// The FIRST handler throws — the spine must catch it (NOT crash), treat it as a
		// decline, and let the SECOND handler claim. A bare upgrade listener that let this
		// escape would be an uncaughtException that takes the whole process down.
		server.upgrade(() => {
			order.push('throw')
			throw new Error('upgrade boom')
		})
		server.upgrade((_request, socket) => {
			order.push('claim')
			socket.write(
				'HTTP/1.1 101 Switching Protocols\r\nUpgrade: websocket\r\nConnection: Upgrade\r\n\r\n',
			)
			socket.end()
			return true
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/ws')
		// The server survived and the second handler claimed the socket.
		expect(outcome.claimed).toBe(true)
		expect(outcome.status).toBe(101)
		// Both ran in order — the throw did not abort the fan-out.
		expect(order).toEqual(['throw', 'claim'])
		// The throw was surfaced on the `error` event (symmetric with the request path).
		expect(errors.count).toBe(1)
		expect(errors.calls[0]?.[0]).toBeInstanceOf(Error)
		// The server is still up — a later request still works.
		server.route({ method: 'GET', path: '/ok', handler: (c) => c.text('ok') })
		expect(await (await fetch(`${handle.base}/ok`)).text()).toBe('ok')
	})

	it('a sole throwing handler declines — the socket is destroyed and upgrade fires handled: false', async () => {
		const upgrades = createRecorder<readonly [IncomingMessage, boolean]>()
		const errors = createRecorder<readonly [unknown]>()
		const server = createServer()
		server.emitter.on('upgrade', upgrades.handler)
		server.emitter.on('error', errors.handler)
		// The ONLY handler throws — nothing claims, so the spine must surface the throw on
		// `error`, treat the upgrade as unclaimed, and destroy the un-upgraded socket.
		server.upgrade(() => {
			throw new Error('sole boom')
		})
		const handle = track(await startServer(server))
		const outcome = await upgradeRequest(handle.base, '/nope')
		// No 101 reached the client — the spine destroyed the un-upgraded socket.
		expect(outcome.claimed).toBe(false)
		// The throw surfaced on `error`, and the `upgrade` event reported the unclaimed outcome.
		expect(errors.count).toBe(1)
		expect(errors.calls[0]?.[0]).toBeInstanceOf(Error)
		expect(upgrades.count).toBe(1)
		expect(upgrades.calls[0]?.[1]).toBe(false)
	})
})
